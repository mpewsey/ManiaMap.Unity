191 private void CreateLayers()
192 {
193 CreateLayersContainer();
194 var capacity = LayerPositions.Length;
195
196 // Destroy extra layers.
197 while (Layers.Count > capacity)
198 {
199 var index = Layers.Count - 1;
200 Destroy(Layers[index].gameObject);
201 Layers.RemoveAt(index);
202 }
203
204 // Create missing layers.
205 while (Layers.Count < capacity)
206 {
207 Layers.Add(LayoutMapLayer.Create(this));
208 }
209 }

272 private void DrawMapTiles(Texture2D texture, int z)
273 {
274 foreach (var room in Layout.Rooms.Values)
275 {
276 // If room Z (layer) value is not equal, go to next room.
277 if (room.Position.Z != z)
278 continue;
279
280 var roomState = LayoutState?.RoomStates[room.Id];
281 var cells = room.Template.Cells;
282 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
283 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
284
285 for (int i = 0; i < cells.Rows; i++)
286 {
287 for (int j = 0; j < cells.Columns; j++)
288 {
289 var cell = cells[i, j];
290
291 // If cell it empty, go to next cell.
292 if (cell == null)
293 continue;
294
295 var position = new Vector2DInt(i, j);
296 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
297
298 // If room state is defined and is not visible, go to next cell.
299 if (!isCompletelyVisible && !roomState.IsVisible)
300 continue;
301
302 // Calculate draw position
303 var x = x0 + MapTiles.TileSize.x * j;
304 var y = y0 - MapTiles.TileSize.y * i;
305 var point = new Vector2Int(x, y);
306
307 // Get adjacent cells
308 var north = cells.GetOrDefault(i - 1, j);
309 var south = cells.GetOrDefault(i + 1, j);
310 var west = cells.GetOrDefault(i, j - 1);
311 var east = cells.GetOrDefault(i, j + 1);
312
313 // Get the wall or door tiles
314 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
315 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
316 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
317 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
318 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
319 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
320
321 // Add cell background fill
322 var rect = new RectInt(point, MapTiles.TileSize);
323 var color = isCompletelyVisible ? ColorUtility.ConvertColor(room.Color) : RoomColor;
324 TextureUtility.CompositeFill(texture, color, rect);
325
326 // Superimpose applicable map tiles
327 TextureUtility.DrawImage(texture, northTile, point);
328 TextureUtility.DrawImage(texture, southTile, point);
329 TextureUtility.DrawImage(texture, westTile, point);
330 TextureUtility.DrawImage(texture, eastTile, point);
331 TextureUtility.DrawImage(texture, topTile, point);
332 TextureUtility.DrawImage(texture, bottomTile, point);
333
334 if (isCompletelyVisible)
335 DrawFeatureTiles(texture, cell, point);
336 }
337 }
338 }
339 }

365 private Texture2D GetTile(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
366 {
367 if (Door.ShowDoor(DoorDrawMode, direction) && cell.GetDoor(direction) != null && DoorExists(room, position, direction))
368 return MapTiles.GetTile(MapTileType.GetDoorTileType(direction));
369
370 if (neighbor == null)
371 return MapTiles.GetTile(MapTileType.GetWallTileType(direction));
372
373 return null;
