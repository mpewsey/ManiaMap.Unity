163 public void CreateLayers(Layout layout, LayoutState state)
164 {
165 Initialize(layout, state);
166 var zs = Layout.Rooms.Values.Select(x => x.Position.Z).Distinct().ToList();
167 zs.Sort();
168 EnsureCapacity(zs.Count);
169
170 // Draw the layers.
171 var size = GetTextureSize();
172
173 for (int i = 0; i < Layers.Count; i++)
174 {
175 var layer = Layers[i];
176 layer.Initialize(size, zs[i]);
177 DrawMap(layer.Sprite.texture, layer.Z);
178 }
179 }

185 private void EnsureCapacity(int capacity)
186 {
187 CreateLayersContainer();
188
189 // Destroy extra layers.
190 while (Layers.Count > capacity)
191 {
192 var index = Layers.Count - 1;
193 Destroy(Layers[index].gameObject);
194 Layers.RemoveAt(index);
195 }
196
197 // Create missing layers.
198 while (Layers.Count < capacity)
199 {
200 Layers.Add(LayoutMapLayer.Create(this));
201 }
202 }

265 private void DrawMapTiles(Texture2D texture, int z)
266 {
267 foreach (var room in Layout.Rooms.Values)
268 {
269 // If room Z (layer) value is not equal, go to next room.
270 if (room.Position.Z != z)
271 continue;
272
273 var roomState = LayoutState?.RoomStates[room.Id];
274 var cells = room.Template.Cells;
275 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
276 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
277
278 for (int i = 0; i < cells.Rows; i++)
279 {
280 for (int j = 0; j < cells.Columns; j++)
281 {
282 var cell = cells[i, j];
283 var position = new Vector2DInt(i, j);
284
285 // If cell it empty, go to next cell.
286 if (cell == null)
287 continue;
288
289 // If room state is defined and is not visible, go to next cell.
290 if (roomState != null && !roomState.IsVisible && !roomState.CellIsVisible(position))
291 continue;
292
293 // Calculate draw position
294 var x = x0 + MapTiles.TileSize.x * j;
295 var y = y0 - MapTiles.TileSize.y * i;
296 var point = new Vector2Int(x, y);
297
298 // Get adjacent cells
299 var north = cells.GetOrDefault(i - 1, j);
300 var south = cells.GetOrDefault(i + 1, j);
301 var west = cells.GetOrDefault(i, j - 1);
302 var east = cells.GetOrDefault(i, j + 1);
303
304 // Get the wall or door tiles
305 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
306 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
307 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
308 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
309 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
310 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
311
312 // Add cell background fill
313 var rect = new RectInt(point, MapTiles.TileSize);
314 var color = roomState == null || roomState.CellIsVisible(position) ? ColorUtility.ConvertColor(room.Color) : RoomColor;
315 TextureUtility.CompositeFill(texture, color, rect);
316
317 // Superimpose applicable map tiles
318 TextureUtility.DrawImage(texture, northTile, point);
319 TextureUtility.DrawImage(texture, southTile, point);
320 TextureUtility.DrawImage(texture, westTile, point);
321 TextureUtility.DrawImage(texture, eastTile, point);
322 TextureUtility.DrawImage(texture, topTile, point);
323 TextureUtility.DrawImage(texture, bottomTile, point);
324
325 DrawFeatureTiles(texture, cell, point);
326 }
327 }
328 }
329 }

355 private Texture2D GetTile(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
356 {
357 if (Door.ShowDoor(DoorDrawMode, direction) && cell.GetDoor(direction) != null && DoorExists(room, position, direction))
358 return MapTiles.GetTile(MapTileType.GetDoorTileType(direction));
359
360 if (neighbor == null)
361 return MapTiles.GetTile(MapTileType.GetWallTileType(direction));
362
363 return null;
364 }
