119 private void SizePages()
120 {
121 while (Pages.Count > PageLayerCoordinates.Count)
122 {
123 var index = Pages.Count - 1;

203 public float SetOnionskinColors(float z, Gradient gradient, float drawDepth = 1)
204 {
205 if (PageLayerCoordinates.Count > 0)
206 {
207 var scale = 0.5f / drawDepth;
208 var minZ = PageLayerCoordinates[0];
209 var maxZ = PageLayerCoordinates[PageLayerCoordinates.Count - 1];
210 z = Mathf.Clamp(z, minZ, maxZ);
211
212 for (int i = 0; i < PageLayerCoordinates.Count; i++)
213 {
214 var t = (PageLayerCoordinates[i] - z) * scale + 0.5f;
215 Pages[i].color = gradient.Evaluate(Mathf.Clamp(t, 0, 1));
216 }
217 }
218
219 return z;
