201 public Graphs.LayoutGraph GetLayoutGraph()

203 var graph = new Graphs.LayoutGraph(Id, Name);

205 foreach (var node in Nodes.OrderBy(x => x.Id))
206 {
207 var other = graph.AddNode(node.Id);

211 other.Color = ConvertColor(node.Color);
212
213 if (!string.IsNullOrWhiteSpace(node.VariationGroup))

215 }
216
217 foreach (var edge in Edges.OrderBy(x => new EdgeIndexes(x.FromNode, x.ToNode)))

219 var other = graph.AddEdge(edge.FromNode, edge.ToNode);

222 other.DoorCode = edge.DoorCode;

264 public bool Paginate(Vector2 spacing)
265 {
266 var result = false;
267 var changed = true;
268
269 for (int k = 0; changed && k < 1000; k++)
270 {
271 changed = false;
272
273 for (int i = 0; i < Nodes.Count; i++)
274 {
275 for (int j = i + 1; j < Nodes.Count; j++)
276 {
277 var iNode = Nodes[i];
278 var jNode = Nodes[j];
279 var delta = jNode.Position - iNode.Position;
280
281 if (Mathf.Abs(delta.x) > spacing.x || Mathf.Abs(delta.y) > spacing.y)
282 continue;
283
284 var sign = new Vector2(Mathf.Sign(delta.x), Mathf.Sign(delta.y));
285 delta = spacing * sign - delta;
286
287 if (Mathf.Abs(delta.x) * spacing.y > Mathf.Abs(delta.y) * spacing.x)
288 delta.x = 0;
289 else
290 delta.y = 0;
291
292 if ((k & 1) == 0)
293 jNode.Position += delta;
294 else
295 iNode.Position -= delta;
296
297 changed = true;
298 }
299 }
300
301 result |= changed;
302 }
303
304 return result;
305 }
