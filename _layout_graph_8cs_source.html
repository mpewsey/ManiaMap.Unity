204 foreach (var node in Nodes.OrderBy(x => x.Id))

277 public bool Paginate(Vector2 spacing)
278 {
279 var result = false;
280 var changed = true;
281
282 for (int k = 0; changed && k < 1000; k++)
283 {
284 changed = false;
285
286 for (int i = 0; i < Nodes.Count; i++)
287 {
288 for (int j = i + 1; j < Nodes.Count; j++)
289 {
290 var iNode = Nodes[i];
291 var jNode = Nodes[j];
292 var delta = jNode.Position - iNode.Position;
293
294 if (Mathf.Abs(delta.x) > spacing.x || Mathf.Abs(delta.y) > spacing.y)
295 continue;
296
297 var sign = new Vector2(Mathf.Sign(delta.x), Mathf.Sign(delta.y));
298 delta = spacing * sign - delta;
299
300 if (Mathf.Abs(delta.x) * spacing.y > Mathf.Abs(delta.y) * spacing.x)
301 delta.x = 0;
302 else
303 delta.y = 0;
304
305 if ((k & 1) == 0)
306 jNode.Position += delta;
307 else
308 iNode.Position -= delta;
309
310 changed = true;
311 }
312 }
313
314 result |= changed;
315 }
316
317 return result;
318 }
