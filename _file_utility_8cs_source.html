20 public static void CreateDirectory(string path)
21 {
22 if (AssetDatabase.IsValidFolder(path))
23 return;
24
25 var separators = new char[] { Path.DirectorySeparatorChar, Path.AltDirectorySeparatorChar };
26 var directories = path.Split(separators);
27
28 if (directories.Length == 0)
29 throw new ArgumentException($"Path is empty: {path}.");
30 if (directories[0] != "Assets")
31 throw new ArgumentException($"Path does not begin with Assets directory: {path}.");
32
33 for (int i = 1; i < directories.Length; i++)
34 {
35 var directory = Path.Combine(directories.Take(i + 1).ToArray());
36
37 if (!AssetDatabase.IsValidFolder(directory))
38 {
39 var root = Path.Combine(directories.Take(i).ToArray());
40 AssetDatabase.CreateFolder(root, directories[i]);
41 }
42 }
43 }

50 public static string ReplaceInvalidFileNameCharacters(string str, char replacement = '_')
51 {
52 var characters = Path.GetInvalidFileNameChars();
53
54 // Make sure replacement character itself is valid.
55 if (characters.Contains(replacement))
56 throw new ArgumentException($"Invalid replacement character: {replacement}.");
57
58 var builder = new StringBuilder(str);
59
60 foreach (var character in characters)
61 {
62 builder.Replace(character, replacement);
63 }
64
