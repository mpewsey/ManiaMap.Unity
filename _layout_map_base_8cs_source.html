134 protected void DrawTiles(Texture2D texture, int z, Padding padding, Color32 backgroundColor)
135 {
136 var bounds = LayoutPack.LayoutBounds;
137 TextureUtility.Fill(texture, backgroundColor);
138 TextureUtility.TileImage(texture, MapTileSet.GetTexture(MapTileType.Grid));
139
140 foreach (var room in LayoutPack.GetRoomsInLayer(z))
141 {
142 var roomState = LayoutPack.LayoutState?.RoomStates[room.Id];
143 var cells = room.Template.Cells;
144 var x0 = (room.Position.Y - bounds.X + padding.Left) * MapTileSet.TileSize.x;
145 var y0 = (bounds.Height + padding.Bottom - room.Position.X + bounds.Y - 1) * MapTileSet.TileSize.y;
146
147 for (int i = 0; i < cells.Rows; i++)
148 {
149 for (int j = 0; j < cells.Columns; j++)
150 {
151 var cell = cells[i, j];
152
153 // If cell it empty, go to next cell.
154 if (cell == null)
155 continue;
156
157 var position = new Vector2DInt(i, j);
158 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
159
160 // If room state is defined and is not visible, go to next cell.
161 if (!isCompletelyVisible && !roomState.IsVisible)
162 continue;
163
164 // Calculate draw position
165 var x = x0 + MapTileSet.TileSize.x * j;
166 var y = y0 - MapTileSet.TileSize.y * i;
167 var point = new Vector2Int(x, y);
168
169 // Get adjacent cells
170 var north = cells.GetOrDefault(i - 1, j);
171 var south = cells.GetOrDefault(i + 1, j);
172 var west = cells.GetOrDefault(i, j - 1);
173 var east = cells.GetOrDefault(i, j + 1);
174
175 // Get the wall or door tiles
176 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
177 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
178 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
179 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
180 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
181 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
182
183 // Add cell background fill
184 var rect = new RectInt(point, MapTileSet.TileSize);
185 var color = isCompletelyVisible ? ColorUtility.ConvertColor4ToColor32(room.Color) : RoomColor;
186 TextureUtility.CompositeFill(texture, color, rect);
187
188 // Superimpose applicable map tiles
189 TextureUtility.DrawImage(texture, northTile, point);
190 TextureUtility.DrawImage(texture, southTile, point);
191 TextureUtility.DrawImage(texture, westTile, point);
192 TextureUtility.DrawImage(texture, eastTile, point);
193 TextureUtility.DrawImage(texture, topTile, point);
194 TextureUtility.DrawImage(texture, bottomTile, point);
195
196 if (isCompletelyVisible)
197 DrawFeatureTiles(texture, cell, point);
198 }
199 }
200 }
201
202 texture.Apply();
203 }

245 var east = cells.GetOrDefault(i, j + 1);
246
247 // Accumulate map tile types
248 var flags = GetTileFlag(room, cell, null, position, DoorDirection.Top);
249 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
250 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
251 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
252 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
253 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
254
255 if (isCompletelyVisible)
256 flags |= GetFeatureFlags(cell);
257
258 // Set the map tile.
259 var color = isCompletelyVisible ? ColorUtility.ConvertColor4ToColor32(room.Color) : RoomColor;
260 var tile = MapTileSet.GetTile(flags, color);
261 tilemap.SetTile(point, tile);
262 }
