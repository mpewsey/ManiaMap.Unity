79 protected void DrawTiles(Texture2D texture, int z, Padding padding, Color32 backgroundColor)
80 {
81 var bounds = LayoutPack.LayoutBounds;
82 TextureUtility.Fill(texture, backgroundColor);
83 TextureUtility.TileImage(texture, MapTileSet.GetTexture(MapTileType.Grid));
84
85 foreach (var room in LayoutPack.GetRoomsInLayer(z))
86 {
87 var roomState = LayoutPack.LayoutState?.RoomStates[room.Id];
88 var cells = room.Template.Cells;
89 var x0 = (room.Position.Y - bounds.X + padding.Left) * MapTileSet.TileSize.x;
90 var y0 = (bounds.Height + padding.Bottom - room.Position.X + bounds.Y - 1) * MapTileSet.TileSize.y;
91
92 for (int i = 0; i < cells.Rows; i++)
93 {
94 for (int j = 0; j < cells.Columns; j++)
95 {
96 var cell = cells[i, j];
97
98 // If cell it empty, go to next cell.
99 if (cell == null)
100 continue;
101
102 var position = new Vector2DInt(i, j);
103 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
104
105 // If room state is defined and is not visible, go to next cell.
106 if (!isCompletelyVisible && !roomState.IsVisible)
107 continue;
108
109 // Calculate draw position
110 var x = x0 + MapTileSet.TileSize.x * j;
111 var y = y0 - MapTileSet.TileSize.y * i;
112 var point = new Vector2Int(x, y);
113
114 // Get adjacent cells
115 var north = cells.GetOrDefault(i - 1, j);
116 var south = cells.GetOrDefault(i + 1, j);
117 var west = cells.GetOrDefault(i, j - 1);
118 var east = cells.GetOrDefault(i, j + 1);
119
120 // Get the wall or door tiles
121 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
122 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
123 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
124 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
125 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
126 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
127
128 // Add cell background fill
129 var rect = new RectInt(point, MapTileSet.TileSize);
130 var color = isCompletelyVisible ? ColorUtility.ConvertColor4ToColor32(room.Color) : RoomColor;
131 TextureUtility.CompositeFill(texture, color, rect);
132
133 // Superimpose applicable map tiles
134 TextureUtility.DrawImage(texture, northTile, point);
135 TextureUtility.DrawImage(texture, southTile, point);
136 TextureUtility.DrawImage(texture, westTile, point);
137 TextureUtility.DrawImage(texture, eastTile, point);
138 TextureUtility.DrawImage(texture, topTile, point);
139 TextureUtility.DrawImage(texture, bottomTile, point);
140
141 if (isCompletelyVisible)
142 DrawFeatureTiles(texture, cell, point);
143 }
144 }
145 }
146
147 texture.Apply();
148 }

178 var y = -room.Position.X - i;

181 // Get adjacent cells
182 var north = cells.GetOrDefault(i - 1, j);
183 var south = cells.GetOrDefault(i + 1, j);
184 var west = cells.GetOrDefault(i, j - 1);
185 var east = cells.GetOrDefault(i, j + 1);
186
187 // Accumulate map tile types
188 var flags = GetTileFlag(room, cell, null, position, DoorDirection.Top);
189 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
190 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
191 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
192 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
193 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
194
195 if (isCompletelyVisible)
196 flags |= GetFeatureFlags(cell);
197
198 // Set the map tile.
199 var color = isCompletelyVisible ? ColorUtility.ConvertColor4ToColor32(room.Color) : RoomColor;
200 var tile = MapTileSet.GetTile(flags, color);
201 tilemap.SetTile(point, tile);
202 }
203 }
