241 private void CreateLayers()
242 {
243 CreateGrid();
244 var capacity = LayerPositions.Length;
245
246 // Destroy extra layers.
247 while (Layers.Count > capacity)
248 {
249 var index = Layers.Count - 1;
250 Destroy(Layers[index].gameObject);

254 // Create missing layers.
255 while (Layers.Count < capacity)

266 private void DrawMap(Tilemap tilemap, int z)
267 {
268 tilemap.ClearAllTiles();
269
270 foreach (var room in Layout.Rooms.Values)
271 {
272 // If room Z (layer) value is not equal, go to next room.
273 if (room.Position.Z != z)
274 continue;
275
276 var roomState = LayoutState?.RoomStates[room.Id];
277 var cells = room.Template.Cells;
278
279 for (int i = 0; i < cells.Rows; i++)
280 {
281 for (int j = 0; j < cells.Columns; j++)
282 {
283 var cell = cells[i, j];
284
285 // If cell it empty, go to next cell.
286 if (cell == null)
287 continue;
288
289 var position = new Vector2DInt(i, j);
290 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
291
292 // If room state is defined and is not visible, go to next cell.
293 if (!isCompletelyVisible && !roomState.IsVisible)
294 continue;
295
296 // Calculate draw position
297 var x = room.Position.Y + j;
298 var y = -room.Position.X - i;
299 var point = new Vector3Int(x, y, 0);
300
301 // Get adjacent cells
302 var north = cells.GetOrDefault(i - 1, j);
303 var south = cells.GetOrDefault(i + 1, j);
304 var west = cells.GetOrDefault(i, j - 1);
305 var east = cells.GetOrDefault(i, j + 1);
306
307 // Accumulate map tile types
308 var flags = GetTileFlag(room, cell, null, position, DoorDirection.Top);
309 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
310 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
311 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
312 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
313 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
314
315 if (isCompletelyVisible)
316 flags |= GetFeatureFlags(cell);
317
318 // Set the map tile.
319 var color = isCompletelyVisible ? ColorUtility.ConvertColor(room.Color) : RoomColor;
320 var tile = MapTilePool.GetTile(flags, color);
321 tilemap.SetTile(point, tile);
322 }
323 }
324 }
325 }

351 private long GetTileFlag(Room room, Cell cell, Cell neighbor, Vector2DInt position, DoorDirection direction)
352 {

354 return MapTilePool.GetFeatureFlag(MapTileType.GetDoorTileType(direction));
355
356 if (neighbor == null)
357 return MapTilePool.GetFeatureFlag(MapTileType.GetWallTileType(direction));
358
