182 private void CreateLayers()
183 {
184 CreateGrid();
185 var capacity = LayerPositions.Length;
186
187 // Destroy extra layers.
188 while (Layers.Count > capacity)
189 {
190 var index = Layers.Count - 1;
191 Destroy(Layers[index].gameObject);
192 Layers.RemoveAt(index);

194
195 // Create missing layers.
196 while (Layers.Count < capacity)

198 Layers.Add(LayoutTilemapLayer.Create(this));

207 private void DrawMap(Tilemap tilemap, int z)
208 {
209 tilemap.ClearAllTiles();
210
211 foreach (var room in Layout.Rooms.Values)
212 {
213 // If room Z (layer) value is not equal, go to next room.
214 if (room.Position.Z != z)
215 continue;
216
217 var roomState = LayoutState?.RoomStates[room.Id];
218 var cells = room.Template.Cells;
219
220 for (int i = 0; i < cells.Rows; i++)
221 {
222 for (int j = 0; j < cells.Columns; j++)
223 {
224 var cell = cells[i, j];
225
226 // If cell it empty, go to next cell.
227 if (cell == null)
228 continue;
229
230 var position = new Vector2DInt(i, j);
231 var isCompletelyVisible = roomState == null || roomState.CellIsVisible(position);
232
233 // If room state is defined and is not visible, go to next cell.
234 if (!isCompletelyVisible && !roomState.IsVisible)
235 continue;
236
237 // Calculate draw position
238 var x = room.Position.Y + j;
239 var y = -room.Position.X - i;
240 var point = new Vector3Int(x, y, 0);
241
242 // Get adjacent cells
243 var north = cells.GetOrDefault(i - 1, j);
244 var south = cells.GetOrDefault(i + 1, j);
245 var west = cells.GetOrDefault(i, j - 1);
246 var east = cells.GetOrDefault(i, j + 1);
247
248 // Accumulate map tile types
249 var flags = GetTileFlag(room, cell, null, position, DoorDirection.Top);
250 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
251 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
252 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
253 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
254 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
255
256 if (isCompletelyVisible)
257 flags |= GetFeatureFlags(cell);
258
259 // Set the map tile.
260 var color = isCompletelyVisible ? ColorUtility.ConvertColor(room.Color) : RoomColor;
261 var tile = MapTilePool.GetTile(flags, color);
262 tilemap.SetTile(point, tile);
263 }
264 }
265 }
266 }
