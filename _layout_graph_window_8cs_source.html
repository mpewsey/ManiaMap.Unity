710 private void HandlePlotAreaEvent()
711 {
712 var size = new Vector2(position.width - GetInspectorWidth(), position.height - Settings.MenuHeight);
713 var rect = new Rect(PlotScrollPosition, size);
714
715 if (rect.Contains(Event.current.mousePosition))
716 {
717 switch (Event.current.type)
718 {
719 case EventType.MouseDown when Event.current.button == LeftMouseButton:
720 DeselectAll();
721 BeginDragSelect();
722 GUI.FocusControl(null);
723 Event.current.Use();
724 break;
725 case EventType.MouseDown when Event.current.button == RightMouseButton:
726 ShowAreaContextMenu();
727 GUI.FocusControl(null);
728 Event.current.Use();
729 break;
730 case EventType.MouseUp when Event.current.button == LeftMouseButton && ActiveTool == Tool.DragSelect:
731 SelectDraggedElements();
732 GUI.FocusControl(null);
733 Event.current.Use();
734 break;
735 case EventType.MouseUp when Event.current.button == LeftMouseButton:
736 ActiveTool = Tool.None;
737 Multiselecting = false;
738 GUI.FocusControl(null);
739 Event.current.Use();
740 break;
741 }
742 }
743 }

805 private void HandleKeyEvent()
806 {
807 if (Event.current.type == EventType.KeyDown)
808 {
809 switch (Event.current.keyCode)
810 {
811 case KeyCode.Escape:
812 DeselectAll();
813 Multiselecting = false;
814 GUI.FocusControl(null);
815 Event.current.Use();

817 case KeyCode.Delete:
818 DeleteSelected();

822 break;
823 case KeyCode.A when Event.current.control:
824 SelectAll();
825 Multiselecting = false;
826 GUI.FocusControl(null);
827 Event.current.Use();
828 break;
829 case KeyCode.RightArrow:
830 case KeyCode.D:

832 GUI.FocusControl(null);
833 Event.current.Use();
834 break;
835 case KeyCode.LeftArrow:

840 break;
