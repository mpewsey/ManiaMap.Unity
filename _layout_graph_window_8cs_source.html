712 private void HandlePlotAreaEvent()
713 {
714 var size = new Vector2(position.width - GetInspectorWidth(), position.height - Settings.MenuHeight);
715 var rect = new Rect(PlotScrollPosition, size);
716
717 if (rect.Contains(Event.current.mousePosition))
718 {
719 switch (Event.current.type)
720 {
721 case EventType.MouseDown when Event.current.button == LeftMouseButton:
722 DeselectAll();
723 BeginDragSelect();
724 GUI.FocusControl(null);
725 Event.current.Use();
726 break;
727 case EventType.MouseDown when Event.current.button == RightMouseButton:
728 ShowAreaContextMenu();
729 GUI.FocusControl(null);
730 Event.current.Use();
731 break;
732 case EventType.MouseUp when Event.current.button == LeftMouseButton && ActiveTool == Tool.DragSelect:
733 SelectDraggedElements();
734 GUI.FocusControl(null);
735 Event.current.Use();
736 break;
737 case EventType.MouseUp when Event.current.button == LeftMouseButton:
738 ActiveTool = Tool.None;
739 Multiselecting = false;
740 GUI.FocusControl(null);
741 Event.current.Use();
742 break;
743 }
744 }
745 }

808 {

811 switch (Event.current.keyCode)
