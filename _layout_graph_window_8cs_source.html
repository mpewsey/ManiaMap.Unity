711 private void HandlePlotAreaEvent()
712 {
713 var size = new Vector2(position.width - GetInspectorWidth(), position.height - Settings.MenuHeight);
714 var rect = new Rect(PlotScrollPosition, size);
715
716 if (rect.Contains(Event.current.mousePosition))
717 {
718 switch (Event.current.type)
719 {
720 case EventType.MouseDown when Event.current.button == LeftMouseButton:
721 DeselectAll();
722 BeginDragSelect();
723 GUI.FocusControl(null);
724 Event.current.Use();
725 break;
726 case EventType.MouseDown when Event.current.button == RightMouseButton:
727 ShowAreaContextMenu();
728 GUI.FocusControl(null);
729 Event.current.Use();
730 break;
731 case EventType.MouseUp when Event.current.button == LeftMouseButton && ActiveTool == Tool.DragSelect:
732 SelectDraggedElements();
733 GUI.FocusControl(null);
734 Event.current.Use();
735 break;
736 case EventType.MouseUp when Event.current.button == LeftMouseButton:
737 ActiveTool = Tool.None;
738 Multiselecting = false;
739 GUI.FocusControl(null);
740 Event.current.Use();
741 break;
742 }
743 }
744 }

752 switch (Event.current.type)

806 private void HandleKeyEvent()
807 {
808 if (Event.current.type == EventType.KeyDown)
809 {
810 switch (Event.current.keyCode)
811 {
812 case KeyCode.Escape:
813 DeselectAll();
814 Multiselecting = false;
815 GUI.FocusControl(null);

818 case KeyCode.Delete:
819 DeleteSelected();
820 Multiselecting = false;

823 break;
824 case KeyCode.A when Event.current.control:
825 SelectAll();
826 Multiselecting = false;
827 GUI.FocusControl(null);
828 Event.current.Use();
829 break;
830 case KeyCode.RightArrow:
831 case KeyCode.D:

833 GUI.FocusControl(null);
834 Event.current.Use();
835 break;

841 break;
