94 public void AddFeature(string name)
95 {
96 if (string.IsNullOrWhiteSpace(name))
97 throw new System.ArgumentException($"Invalid feature name: {name}.");
98
99 if (FeatureFlags.ContainsKey(name))
100 return;
101
102 if (FeatureFlags.Count >= MaxFeatureCount)
103 throw new System.ArgumentException($"Feature count exceeded. Cannot add feature: {name}.");
104
105 var flag = 1 << FeatureFlags.Count;
106 FeatureFlags.Add(name, flag);
107 FeatureNames.Add(flag, name);
108 }
