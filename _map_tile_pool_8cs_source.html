87 public void AddFeature(string name)
88 {
89 if (string.IsNullOrWhiteSpace(name))
90 throw new System.ArgumentException($"Invalid feature name: {name}.");
91
92 if (FeatureFlags.ContainsKey(name))
93 return;
94
95 if (FeatureFlags.Count >= MaxFeatureCount)
96 throw new System.ArgumentException($"Feature count exceeded. Cannot add feature: {name}.");
97
98 var flag = 1 << FeatureFlags.Count;
99 FeatureFlags.Add(name, flag);
100 FeatureNames.Add(flag, name);
101 }

134 public Tile GetTile(long flags, Color32 color)
