55 private static void ProcessEditCells(SceneView sceneView)
56 {
57 if (!DisplayToolbar || !DisplayCells || CellEditMode == CellActivity.None || Room == null || !Event.current.isMouse)
58 return;
59
60 if (Event.current.type == EventType.MouseDown && Event.current.button == LeftMouseButton)
61 {
62 if (!MouseButtonPressed)
63 MouseButtonDownIndex = Room.GlobalPositionToCellIndex(GetMousePosition(sceneView));
64
65 MouseButtonPressed = true;
66 Event.current.Use();
67 return;
68 }
69
70 if (MouseButtonPressed)
71 {
72 var startIndex = MouseButtonDownIndex;
73 var endIndex = Room.GlobalPositionToCellIndex(GetMousePosition(sceneView));
74
75 if (Room.SetCellActivities(startIndex, endIndex, CellEditMode))
76 EditorUtility.SetDirty(Room);
77 }
78
79 MouseButtonPressed = false;
80 }
