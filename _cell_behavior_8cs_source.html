190 public static CellBehavior FindClosestCell(Transform transform)
191 {
192 var room = transform.GetComponentInParent<RoomBehavior>();
193
194 if (room == null)
195 throw new ArgumentException($"Parent room not found for transform: {transform}.");
196
197 var cells = room.GetNonEmptyCells();
198 var distances = CellSqrDistances(cells, transform.position);
199 var index = Maths.MinIndex(distances);
200 return index < 0 ? null : cells[index];
201 }

208 private static float[] CellSqrDistances(List<CellBehavior> cells, Vector3 position)
209 {
210 var distances = new float[cells.Count];
211
212 for (int i = 0; i < distances.Length; i++)
213 {
214 var delta = cells[i].transform.position - position;
215 distances[i] = delta.sqrMagnitude;
216 }
217
218 return distances;
219 }
