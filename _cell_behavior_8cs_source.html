201 public static CellBehavior FindClosestCell(Transform transform)
202 {
203 var room = transform.GetComponentInParent<RoomBehavior>();
204
205 if (room == null)
206 throw new ArgumentException($"Parent room not found for transform: {transform}.");
207
208 var cells = room.GetNonEmptyCells();
209 var distances = CellSqrDistances(cells, transform.position);
210 var index = Maths.MinIndex(distances);
211 return index < 0 ? null : cells[index];
212 }

219 private static float[] CellSqrDistances(List<CellBehavior> cells, Vector3 position)
220 {
221 var distances = new float[cells.Count];
222
223 for (int i = 0; i < distances.Length; i++)
224 {
225 var delta = cells[i].transform.position - position;
226 distances[i] = delta.sqrMagnitude;
227 }
228
229 return distances;
230 }
