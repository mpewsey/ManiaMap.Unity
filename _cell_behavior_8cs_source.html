215 public static CellBehavior FindClosestCell(Transform transform)
216 {
217 var room = transform.GetComponentInParent<RoomBehavior>();
218
219 if (room == null)
220 throw new ArgumentException($"Parent room not found for transform: {transform}.");
221
222 var cells = room.GetNonEmptyCells();
223 var distances = CellSqrDistances(cells, transform.position);
224 var min = Mathf.Min(distances);
225
226 for (int i = 0; i < distances.Length; i++)
227 {
228 if (distances[i] == min)
229 return cells[i];
230 }
231
232 return null;
233 }

240 private static float[] CellSqrDistances(List<CellBehavior> cells, Vector3 position)
241 {
242 var distances = new float[cells.Count];
243
244 for (int i = 0; i < distances.Length; i++)
245 {
246 var delta = cells[i].transform.position - position;
247 distances[i] = delta.sqrMagnitude;
248 }
249
250 return distances;
251 }
