119 public ManiaMap.Cell GetCell()
120 {
121 var doors = FindDoors();
122 var spots = FindCollectableSpots();
123
124 if (IsEmpty)
125 {
126 if (doors.Count > 0)
127 throw new EmptyCellException($"Doors assigned to empty cell: {Index}.");
128 if (spots.Count > 0)
129 throw new EmptyCellException($"Collectable spots assigned to empty cell: {Index}.");
130 return ManiaMap.Cell.Empty;
131 }
132
133 var cell = ManiaMap.Cell.New;
134
135 // Add doors.
136 foreach (var door in doors)
137 {
138 if (cell.GetDoor(door.Direction) != null)
139 throw new DuplicateDirectionException($"Door direction already exists: Index = {Index}, Direction = {door.Direction}.");
140 cell.SetDoor(door.Direction, door.GetDoor());
141 }
142
143 // Add collectable spots.
144 foreach (var spot in spots)
145 {
146 if (spot.Group == null)
147 throw new UnassignedCollectableGroupException($"Collectable group not assigned to collectable spot: {spot.name}.");
148 cell.AddCollectableSpot(spot.Id, spot.Group.Name);
149 }
150
151 return cell;
152 }

159 public static Cell FindClosestCell(Transform transform)
160 {
161 Cell closest = null;
162 var minDistance = float.PositiveInfinity;
163 var room = transform.GetComponentInParent<Room>();
164
165 if (room == null)
166 throw new ArgumentException($"Parent room not found for transform: {transform.name}.");
167
168 for (int i = 0; i < room.Size.x; i++)
169 {
170 for (int j = 0; j < room.Size.y; j++)
171 {
172 var cell = room.GetCell(i, j);
173
174 if (cell.IsEmpty)
175 continue;
176
177 var delta = cell.transform.position - transform.position;
178 var distance = delta.sqrMagnitude;
179
180 if (distance < minDistance)
181 {
182 closest = cell;
183 minDistance = distance;
184 }
185 }
186 }
187
188 return closest;
189 }
