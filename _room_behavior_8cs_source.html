180 public static AsyncOperationHandle<GameObject> InstantiateRoomAsync(Uid id,

182 RoomPositionOption position = RoomPositionOption.UseManagerSetting)
183 {

187 var layoutState = manager.LayoutState;
188 var roomLayout = layout.Rooms[id];

197 };

311 public Vector3 Swizzle(Vector3 vector)
312 {
313 switch (CellPlane)
314 {
315 case CellPlane.XY:
316 return new Vector3(vector.x, vector.y, -vector.z);
317 case CellPlane.XZ:
318 return new Vector3(vector.x, vector.z, vector.y);
319 default:
320 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
321 }
322 }

329 public Vector3 InverseSwizzle(Vector3 vector)

331 switch (CellPlane)

346 public Vector2Int GetCellIndex(Vector3 position)

415 public void CreateCells()
416 {
417 CreateCellContainer();
418
419 // Destroy extra rows.
420 while (CellContainer.childCount > Size.x)
421 {
422 var row = CellContainer.GetChild(CellContainer.childCount - 1);
423 DestroyImmediate(row.gameObject);
424 }
425
426 // Create new rows.
427 while (CellContainer.childCount < Size.x)
428 {
429 var obj = new GameObject("<New Row>");
430 obj.transform.SetParent(CellContainer);
431 }
432
433 for (int i = 0; i < CellContainer.childCount; i++)
434 {
435 CreateRow(i);
436 }
437 }
438
444 private void CreateRow(int row)
445 {
446 var container = CellContainer.GetChild(row);
447 container.name = $"<Row {row}>";
448
449 // Destroy extra cells.
450 while (container.childCount > Size.y)
451 {
452 var cell = container.GetChild(container.childCount - 1);
453 DestroyImmediate(cell.gameObject);
454 }
455
456 // Create new cells.
457 while (container.childCount < Size.y)
458 {
459 var obj = new GameObject("<New Cell>");
460 obj.transform.SetParent(container);
461 obj.AddComponent<CellBehavior>();
462 }
463
464 for (int j = 0; j < container.childCount; j++)
465 {
466 var index = new Vector2Int(row, j);
467 var cell = container.GetChild(j).GetComponent<CellBehavior>();
468 cell.Initialize(this, index);
469 }
470 }

475 public RoomTemplate CreateData()
476 {
477 AutoAssign();
478 var cells = new Array2D<Cell>(Size.x, Size.y);
479
480 for (int i = 0; i < cells.Rows; i++)
481 {
482 for (int j = 0; j < cells.Columns; j++)
483 {
484 cells[i, j] = GetCell(i, j).CreateData();
485 }
486 }
487
