239 public Vector3 Swizzle(Vector3 vector)
240 {
241 switch (CellPlane)
242 {
243 case Plane.XY:
244 return new Vector3(vector.x, vector.y, -vector.z);
245 case Plane.XZ:
246 return new Vector3(vector.x, vector.z, vector.y);
247 default:
248 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
249 }
250 }

257 public Vector3 InverseSwizzle(Vector3 vector)
258 {
259 switch (CellPlane)

264 return new Vector3(vector.x, vector.z, vector.y);

266 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");

343 public void CreateCells()
344 {
345 CreateCellContainer();
346
347 // Destroy extra rows.
348 while (CellContainer.childCount > Size.x)
349 {
350 var row = CellContainer.GetChild(CellContainer.childCount - 1);
351 DestroyImmediate(row.gameObject);
352 }
353
354 // Create new rows.
355 while (CellContainer.childCount < Size.x)
356 {
357 var obj = new GameObject("<New Row>");
358 obj.transform.SetParent(CellContainer);
359 }
360
361 for (int i = 0; i < CellContainer.childCount; i++)
362 {
363 CreateRow(i);
364 }
365 }

372 private void CreateRow(int row)
373 {
374 var container = CellContainer.GetChild(row);
375 container.name = $"<Row {row}>";
376
377 // Destroy extra cells.
378 while (container.childCount > Size.y)
379 {
380 var cell = container.GetChild(container.childCount - 1);
381 DestroyImmediate(cell.gameObject);
382 }
383
384 // Create new cells.
385 while (container.childCount < Size.y)
386 {
387 var obj = new GameObject("<New Cell>");
388 obj.transform.SetParent(container);
389 obj.AddComponent<CellBehavior>();
390 }
391
392 for (int j = 0; j < container.childCount; j++)
393 {
394 var index = new Vector2Int(row, j);
395 var cell = container.GetChild(j).GetComponent<CellBehavior>();
396 cell.Initialize(this, index);
397 }
