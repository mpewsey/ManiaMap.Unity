309 public Vector3 Swizzle(Vector3 vector)
310 {
311 switch (CellPlane)
312 {
313 case CellPlane.XY:
314 return new Vector3(vector.x, vector.y, -vector.z);
315 case CellPlane.XZ:
316 return new Vector3(vector.x, vector.z, vector.y);
317 default:
318 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
319 }
320 }

413 public void CreateCells()
414 {
415 CreateCellContainer();
416
417 // Destroy extra rows.
418 while (CellContainer.childCount > Size.x)
419 {
420 var row = CellContainer.GetChild(CellContainer.childCount - 1);
421 DestroyImmediate(row.gameObject);
422 }
423
424 // Create new rows.
425 while (CellContainer.childCount < Size.x)
426 {
427 var obj = new GameObject("<New Row>");
428 obj.transform.SetParent(CellContainer);
429 }
430
431 for (int i = 0; i < CellContainer.childCount; i++)
432 {
433 CreateRow(i);
434 }
435 }
436
442 private void CreateRow(int row)
443 {
444 var container = CellContainer.GetChild(row);
445 container.name = $"<Row {row}>";
446
447 // Destroy extra cells.
448 while (container.childCount > Size.y)
449 {
450 var cell = container.GetChild(container.childCount - 1);
451 DestroyImmediate(cell.gameObject);
452 }
453
454 // Create new cells.
455 while (container.childCount < Size.y)
456 {
457 var obj = new GameObject("<New Cell>");
458 obj.transform.SetParent(container);
459 obj.AddComponent<CellBehavior>();
460 }
461
462 for (int j = 0; j < container.childCount; j++)
463 {
464 var index = new Vector2Int(row, j);
465 var cell = container.GetChild(j).GetComponent<CellBehavior>();
466 cell.Initialize(this, index);
467 }
