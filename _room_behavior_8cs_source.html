308 public Vector3 Swizzle(Vector3 vector)
309 {
310 switch (CellPlane)
311 {
312 case CellPlane.XY:
313 return new Vector3(vector.x, vector.y, -vector.z);
314 case CellPlane.XZ:
315 return new Vector3(vector.x, vector.z, vector.y);
316 default:
317 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
318 }
319 }

412 public void CreateCells()
413 {
414 CreateCellContainer();
415
416 // Destroy extra rows.
417 while (CellContainer.childCount > Size.x)
418 {
419 var row = CellContainer.GetChild(CellContainer.childCount - 1);
420 DestroyImmediate(row.gameObject);
421 }
422
423 // Create new rows.
424 while (CellContainer.childCount < Size.x)
425 {
426 var obj = new GameObject("<New Row>");
427 obj.transform.SetParent(CellContainer);
428 }
429
430 for (int i = 0; i < CellContainer.childCount; i++)
431 {
432 CreateRow(i);
433 }
434 }
435
441 private void CreateRow(int row)
442 {
443 var container = CellContainer.GetChild(row);
444 container.name = $"<Row {row}>";
445
446 // Destroy extra cells.
447 while (container.childCount > Size.y)
448 {
449 var cell = container.GetChild(container.childCount - 1);
450 DestroyImmediate(cell.gameObject);
451 }
452
453 // Create new cells.
454 while (container.childCount < Size.y)
455 {
456 var obj = new GameObject("<New Cell>");
457 obj.transform.SetParent(container);
458 obj.AddComponent<CellBehavior>();
459 }
460
461 for (int j = 0; j < container.childCount; j++)
462 {
463 var index = new Vector2Int(row, j);
464 var cell = container.GetChild(j).GetComponent<CellBehavior>();
465 cell.Initialize(this, index);
466 }
467 }
