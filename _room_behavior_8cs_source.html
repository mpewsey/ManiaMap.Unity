275 public Vector3 Swizzle(Vector3 vector)
276 {
277 switch (CellPlane)
278 {
279 case CellPlane.XY:
280 return new Vector3(vector.x, vector.y, -vector.z);
281 case CellPlane.XZ:
282 return new Vector3(vector.x, vector.z, vector.y);
283 default:
284 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
285 }
286 }

379 public void CreateCells()
380 {
381 CreateCellContainer();
382
383 // Destroy extra rows.
384 while (CellContainer.childCount > Size.x)
385 {
386 var row = CellContainer.GetChild(CellContainer.childCount - 1);
387 DestroyImmediate(row.gameObject);
388 }
389
390 // Create new rows.
391 while (CellContainer.childCount < Size.x)
392 {
393 var obj = new GameObject("<New Row>");
394 obj.transform.SetParent(CellContainer);
395 }
396
397 for (int i = 0; i < CellContainer.childCount; i++)
398 {
399 CreateRow(i);
400 }
401 }
402
408 private void CreateRow(int row)
409 {
410 var container = CellContainer.GetChild(row);
411 container.name = $"<Row {row}>";
412
413 // Destroy extra cells.
414 while (container.childCount > Size.y)
415 {
416 var cell = container.GetChild(container.childCount - 1);
417 DestroyImmediate(cell.gameObject);
418 }
419
420 // Create new cells.
421 while (container.childCount < Size.y)
422 {
423 var obj = new GameObject("<New Cell>");
424 obj.transform.SetParent(container);
425 obj.AddComponent<CellBehavior>();
426 }
427
428 for (int j = 0; j < container.childCount; j++)
429 {
430 var index = new Vector2Int(row, j);
431 var cell = container.GetChild(j).GetComponent<CellBehavior>();
432 cell.Initialize(this, index);
433 }
434 }

439 public RoomTemplate CreateData()
440 {
441 AutoAssign();
442 var cells = new Array2D<Cell>(Size.x, Size.y);
443
444 for (int i = 0; i < cells.Rows; i++)
445 {
