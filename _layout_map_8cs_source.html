125 public void CreateLayers(Layout layout, LayoutState state)
126 {
127 Initialize(layout, state);
128 var zs = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
129 EnsureCapacity(zs.Count);
130
131 // Draw the layers.
132 int i = 0;
133 var size = GetTextureSize();
134 var layers = GetComponentsInChildren<LayoutMapLayer>();
135
136 foreach (var z in zs.OrderBy(x => x))
137 {
138 var layer = layers[i++];
139 layer.Initialize(size, z);
140 DrawMap(layer.Sprite.texture, layer.Z);
141 }
142 }

148 private void EnsureCapacity(int capacity)
149 {
150 CreateLayersContainer();
151
152 // Destroy extra layers.
153 while (LayersContainer.childCount > capacity)
154 {
155 Destroy(LayersContainer.GetChild(LayersContainer.childCount - 1).gameObject);
156 }
157
158 // Create missing layers.
159 while (LayersContainer.childCount < capacity)
160 {
161 LayoutMapLayer.Create(this);
162 }
163 }

226 private void DrawMapTiles(Texture2D texture, int z)
227 {
228 foreach (var room in Layout.Rooms.Values)
229 {
230 // If room Z (layer) value is not equal, go to next room.
231 if (room.Position.Z != z)
232 continue;
233
234 var roomState = LayoutState?.RoomStates[room.Id];
235 var cells = room.Template.Cells;
236 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
237 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
238
239 for (int i = 0; i < cells.Rows; i++)
240 {
241 for (int j = 0; j < cells.Columns; j++)
242 {
243 var cell = cells[i, j];
244 var position = new Vector2DInt(i, j);
245
246 // If cell it empty, go to next cell.
247 if (cell == null)
248 continue;
249
250 // If room state is defined and is not visible, go to next cell.
251 if (roomState != null && !roomState.CellIsVisible(position))
252 continue;
253
254 // Calculate draw position
255 var x = x0 + MapTiles.TileSize.x * j;
256 var y = y0 - MapTiles.TileSize.y * i;
257 var point = new Vector2Int(x, y);
258
259 // Get adjacent cells
260 var north = cells.GetOrDefault(i - 1, j);
261 var south = cells.GetOrDefault(i + 1, j);
262 var west = cells.GetOrDefault(i, j - 1);
263 var east = cells.GetOrDefault(i, j + 1);
264
265 // Get the wall or door tiles
266 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
267 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
268 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
269 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
270 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
271 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
272
273 // Add cell background fill
274 var rect = new RectInt(point, MapTiles.TileSize);
275 TextureUtility.CompositeFill(texture, ColorUtility.ConvertColor(room.Color), rect);
276
277 // Superimpose applicable map tiles
278 TextureUtility.DrawImage(texture, northTile, point);
279 TextureUtility.DrawImage(texture, southTile, point);
280 TextureUtility.DrawImage(texture, westTile, point);
281 TextureUtility.DrawImage(texture, eastTile, point);
282 TextureUtility.DrawImage(texture, topTile, point);
283 TextureUtility.DrawImage(texture, bottomTile, point);
284
285 DrawFeatureTiles(texture, cell, point);
286 }
287 }
288 }

315 private Texture2D GetTile(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
316 {
317 if (cell.GetDoor(direction) != null && DoorExists(room, position, direction))
318 return MapTiles.GetTile(MapTileType.GetDoorTileType(direction));
319
320 if (neighbor == null)
321 return MapTiles.GetTile(MapTileType.GetWallTileType(direction));
322
323 return null;
