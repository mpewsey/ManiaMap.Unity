108 private static byte[] GetImageBytes(Texture2D texture, string ext)
109 {
110 switch (ext.ToLower())
111 {
112 case ".png":
113 return texture.EncodeToPNG();
114 case ".jpg":
115 case ".jpeg":
116 return texture.EncodeToJPG();
117 default:
118 throw new ArgumentException($"Unhandled file extension: {ext}");

127 private void CreateTextures()
128 {
129 var width = MapTiles.TileSize.x * (Padding.Left + Padding.Right + LayoutBounds.Width);
130 var height = MapTiles.TileSize.y * (Padding.Top + Padding.Bottom + LayoutBounds.Height);
131 var layers = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
132 var removeLayers = MapLayers.Keys.Where(x => !layers.Contains(x)).ToList();
133
134 foreach (var z in removeLayers)
135 {
136 Destroy(MapLayers[z]);
137 MapLayers.Remove(z);
138 }
139
140 foreach (var z in layers)
141 {
142 if (!MapLayers.TryGetValue(z, out Texture2D map))
143 MapLayers.Add(z, new Texture2D(width, height));
144 else if (map.width != width || map.height != height)
145 map.Reinitialize(width, height);
146 }
147 }

300 private void DrawMapTiles(Texture2D texture, int z)
301 {
302 foreach (var room in Layout.Rooms.Values)
303 {
304 // If room Z (layer) value is not equal, go to next room.
305 if (room.Position.Z != z)
306 continue;
307
308 var roomState = LayoutState?.RoomStates[room.Id];
309 var cells = room.Template.Cells;
310 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
311 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
312
313 for (int i = 0; i < cells.Rows; i++)
314 {
315 for (int j = 0; j < cells.Columns; j++)
316 {
317 var cell = cells[i, j];
318 var position = new Vector2DInt(i, j);
319
320 // If cell it empty, go to next cell.
321 if (cell == null)
322 continue;
323
324 // If room state is defined and is not visible, go to next cell.
325 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
326 continue;
327
328 // Calculate draw position
329 var x = x0 + MapTiles.TileSize.x * j;
330 var y = y0 - MapTiles.TileSize.y * i;
331 var point = new Vector2Int(x, y);
332
333 // Get adjacent cells
334 var north = cells.GetOrDefault(i - 1, j);
335 var south = cells.GetOrDefault(i + 1, j);
336 var west = cells.GetOrDefault(i, j - 1);
337 var east = cells.GetOrDefault(i, j + 1);
338
339 // Get the wall or door tiles
340 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
341 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
342 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
343 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
344 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
345 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
346
347 // Add cell background fill
348 DrawTileFill(texture, ConvertColor(room.Color), point);
349
350 // Superimpose applicable map tiles
351 DrawMapTile(texture, northTile, point);
352 DrawMapTile(texture, southTile, point);
353 DrawMapTile(texture, westTile, point);
354 DrawMapTile(texture, eastTile, point);
355 DrawMapTile(texture, topTile, point);
356 DrawMapTile(texture, bottomTile, point);
357 }
358 }

447 private static Color CompositeColors(Color colorA, Color colorB)
448 {
449 var alpha1 = colorA.a;
450 var alpha2 = colorB.a * (1 - colorA.a);
451 var alpha = alpha1 + alpha2;
452 alpha1 /= alpha;
453 alpha2 /= alpha;
454 var red = colorA.r * alpha1 + colorB.r * alpha2;
455 var green = colorA.g * alpha1 + colorB.g * alpha2;
456 var blue = colorA.b * alpha1 + colorB.b * alpha2;
457 return new Color(red, green, blue, alpha);
458 }
