104 private void CreateTextures()
105 {
106 var size = GetTextureSize();
107 var layers = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
108 var removeLayers = Textures.Keys.Where(x => !layers.Contains(x)).ToList();
109
110 foreach (var z in removeLayers)
111 {
112 Textures.Remove(z);
113 }
114
115 foreach (var z in layers)
116 {
117 if (!Textures.TryGetValue(z, out Texture2D map))
118 Textures.Add(z, new Texture2D(size.x, size.y));
119 else if (map.width != size.x || map.height != size.y)
120 map.Reinitialize(size.x, size.y);
121 }
122 }

193 private void DrawMapTiles(Texture2D texture, int z)
194 {
195 foreach (var room in Layout.Rooms.Values)
196 {
197 // If room Z (layer) value is not equal, go to next room.
198 if (room.Position.Z != z)
199 continue;
200
201 var roomState = LayoutState?.RoomStates[room.Id];
202 var cells = room.Template.Cells;
203 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
204 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
205
206 for (int i = 0; i < cells.Rows; i++)
207 {
208 for (int j = 0; j < cells.Columns; j++)
209 {
210 var cell = cells[i, j];
211 var position = new Vector2DInt(i, j);
212
213 // If cell it empty, go to next cell.
214 if (cell == null)
215 continue;
216
217 // If room state is defined and is not visible, go to next cell.
218 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
219 continue;
220
221 // Calculate draw position
222 var x = x0 + MapTiles.TileSize.x * j;
223 var y = y0 - MapTiles.TileSize.y * i;
224 var point = new Vector2Int(x, y);
225
226 // Get adjacent cells
227 var north = cells.GetOrDefault(i - 1, j);
228 var south = cells.GetOrDefault(i + 1, j);
229 var west = cells.GetOrDefault(i, j - 1);
230 var east = cells.GetOrDefault(i, j + 1);
231
232 // Get the wall or door tiles
233 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
234 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
235 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
236 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
237 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
238 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
239
240 // Add cell background fill
241 var rect = new RectInt(point, MapTiles.TileSize);
242 TextureUtility.CompositeFill(texture, ColorUtility.ConvertColor(room.Color), rect);
243
244 // Superimpose applicable map tiles
245 TextureUtility.DrawImage(texture, northTile, point);
246 TextureUtility.DrawImage(texture, southTile, point);
247 TextureUtility.DrawImage(texture, westTile, point);
248 TextureUtility.DrawImage(texture, eastTile, point);
249 TextureUtility.DrawImage(texture, topTile, point);
250 TextureUtility.DrawImage(texture, bottomTile, point);
251 }
252 }
253 }

265 private Texture2D GetTile(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
266 {
