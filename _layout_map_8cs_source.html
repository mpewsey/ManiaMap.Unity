126 public void CreateLayers(Layout layout, LayoutState state)
127 {
128 Initialize(layout, state);
129 var zs = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
130 EnsureCapacity(zs.Count);
131
132 // Draw the layers.
133 int i = 0;
134 var size = GetTextureSize();
135 var layers = GetComponentsInChildren<LayoutMapLayer>();
136
137 foreach (var z in zs.OrderBy(x => x))
138 {
139 var layer = layers[i++];
140 layer.Initialize(size, z);
141 DrawMap(layer.Sprite.texture, layer.Z);
142 }
143 }

149 private void EnsureCapacity(int capacity)
150 {
151 CreateLayersContainer();
152
153 // Destroy extra layers.
154 while (LayersContainer.childCount > capacity)
155 {
156 Destroy(LayersContainer.GetChild(LayersContainer.childCount - 1).gameObject);
157 }
158
159 // Create missing layers.
160 while (LayersContainer.childCount < capacity)
161 {
162 LayoutMapLayer.Create(this);
163 }
164 }

227 private void DrawMapTiles(Texture2D texture, int z)
228 {
229 foreach (var room in Layout.Rooms.Values)
230 {
231 // If room Z (layer) value is not equal, go to next room.
232 if (room.Position.Z != z)
233 continue;
234
235 var roomState = LayoutState?.RoomStates[room.Id];
236 var cells = room.Template.Cells;
237 var x0 = (room.Position.Y - LayoutBounds.X + Padding.Left) * MapTiles.TileSize.x;
238 var y0 = (LayoutBounds.Height + Padding.Bottom - room.Position.X + LayoutBounds.Y - 1) * MapTiles.TileSize.y;
239
240 for (int i = 0; i < cells.Rows; i++)
241 {
242 for (int j = 0; j < cells.Columns; j++)
243 {
244 var cell = cells[i, j];
245 var position = new Vector2DInt(i, j);
246
247 // If cell it empty, go to next cell.
248 if (cell == null)
249 continue;
250
251 // If room state is defined and is not visible, go to next cell.
252 if (roomState != null && !roomState.CellIsVisible(position))
253 continue;
254
255 // Calculate draw position
256 var x = x0 + MapTiles.TileSize.x * j;
257 var y = y0 - MapTiles.TileSize.y * i;
258 var point = new Vector2Int(x, y);
259
260 // Get adjacent cells
261 var north = cells.GetOrDefault(i - 1, j);
262 var south = cells.GetOrDefault(i + 1, j);
263 var west = cells.GetOrDefault(i, j - 1);
264 var east = cells.GetOrDefault(i, j + 1);
265
266 // Get the wall or door tiles
267 var topTile = GetTile(room, cell, null, position, DoorDirection.Top);
268 var bottomTile = GetTile(room, cell, null, position, DoorDirection.Bottom);
269 var northTile = GetTile(room, cell, north, position, DoorDirection.North);
270 var southTile = GetTile(room, cell, south, position, DoorDirection.South);
271 var westTile = GetTile(room, cell, west, position, DoorDirection.West);
272 var eastTile = GetTile(room, cell, east, position, DoorDirection.East);
273
274 // Add cell background fill
275 var rect = new RectInt(point, MapTiles.TileSize);
276 TextureUtility.CompositeFill(texture, ColorUtility.ConvertColor(room.Color), rect);
277
278 // Superimpose applicable map tiles
279 TextureUtility.DrawImage(texture, northTile, point);
280 TextureUtility.DrawImage(texture, southTile, point);
281 TextureUtility.DrawImage(texture, westTile, point);
282 TextureUtility.DrawImage(texture, eastTile, point);
283 TextureUtility.DrawImage(texture, topTile, point);
284 TextureUtility.DrawImage(texture, bottomTile, point);
285
286 DrawFeatureTiles(texture, cell, point);
287 }
288 }
289 }

316 private Texture2D GetTile(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
317 {
318 if (cell.GetDoor(direction) != null && DoorExists(room, position, direction))
319 return MapTiles.GetTile(MapTileType.GetDoorTileType(direction));
320
321 if (neighbor == null)
322 return MapTiles.GetTile(MapTileType.GetWallTileType(direction));
323
324 return null;
