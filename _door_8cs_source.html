151 public void AssignClosestDirection()
152 {
153 if (Cell == null)
154 return;
155
156 var delta = transform.position - Cell.transform.position;
157
158 var distances = new List<float>
159 {
160 Vector3.Dot(delta, Room.Swizzle(Vector2.up)), // North
161 Vector3.Dot(delta, Room.Swizzle(Vector2.down)), // South
162 Vector3.Dot(delta, Room.Swizzle(Vector2.right)), // East
163 Vector3.Dot(delta, Room.Swizzle(Vector2.left)), // West
164 Vector3.Dot(delta, Room.Swizzle(Vector3.forward)), // Top
165 Vector3.Dot(delta, Room.Swizzle(Vector3.back)), // Bottom
166 };
167
168 var max = distances.Max();
169 var index = distances.FindIndex(x => x == max);
170 Direction = (DoorDirection)index;
171 }
