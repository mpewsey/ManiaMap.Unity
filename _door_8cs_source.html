126 public void AssignClosestDirection()
127 {
128 if (Cell == null)
129 return;
130
131 var delta = transform.position - Cell.transform.position;
132
133 var distances = new List<float>
134 {
135 Vector3.Dot(delta, Room.Swizzle(Vector2.up)), // North
136 Vector3.Dot(delta, Room.Swizzle(Vector2.down)), // South
137 Vector3.Dot(delta, Room.Swizzle(Vector2.right)), // East
138 Vector3.Dot(delta, Room.Swizzle(Vector2.left)), // West
139 Vector3.Dot(delta, Room.Swizzle(Vector3.forward)), // Top
140 Vector3.Dot(delta, Room.Swizzle(Vector3.back)), // Bottom
141 };
142
143 var max = distances.Max();
144 var index = distances.FindIndex(x => x == max);
145 Direction = index < 0 ? DoorDirection.North : (DoorDirection)index;
146 }
