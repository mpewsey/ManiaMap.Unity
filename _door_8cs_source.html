169 public void AssignClosestDirection()
170 {
171 if (Cell == null)
172 return;
173
174 var delta = transform.position - Cell.transform.position;
175 var north = Vector3.Dot(delta, Cell.Room.Swizzle(Vector2.up));
176 var south = Vector3.Dot(delta, Cell.Room.Swizzle(Vector2.down));
177 var east = Vector3.Dot(delta, Cell.Room.Swizzle(Vector2.right));
178 var west = Vector3.Dot(delta, Cell.Room.Swizzle(Vector2.left));
179 var top = Vector3.Dot(delta, Cell.Room.Swizzle(Vector3.forward));
180 var bottom = Vector3.Dot(delta, Cell.Room.Swizzle(Vector3.back));
181 var max = Mathf.Max(north, south, east, west, top, bottom);
182
183 if (max == north)
184 Direction = DoorDirection.North;
185 else if (max == south)
186 Direction = DoorDirection.South;
187 else if (max == east)
188 Direction = DoorDirection.East;
189 else if (max == west)
190 Direction = DoorDirection.West;
191 else if (max == top)
192 Direction = DoorDirection.Top;
193 else
194 Direction = DoorDirection.Bottom;
