137 private static void SaveRoomTemplate(string path)
138 {
139 using (var scope = new PrefabUtility.EditPrefabContentsScope(path))
140 {
141 var prefab = scope.prefabContentsRoot;
142
143 if (!prefab.TryGetComponent(out Room room))
144 return;
145
146 Debug.Log($"Processing room at {path}.");
147 var savePath = GetTemplateSavePath(room);
148 var template = AssetDatabase.LoadAssetAtPath<RoomTemplate>(savePath);
149
150 if (template == null)
151 {
152 template = CreateInstance<RoomTemplate>();
153 template.Init(room.GetTemplate());
154 AssetDatabase.CreateAsset(template, savePath);
155 }
156 else
157 {
158 template.Init(room.GetTemplate());
159 EditorUtility.SetDirty(template);
160 AssetDatabase.SaveAssetIfDirty(template);
161 }
162 }
163 }
