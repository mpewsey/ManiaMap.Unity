23 private void DrawPipelineErrorBox()
24 {
25 var inputMessages = new List<string>();
26 inputMessages.Add("Inputs contain errors:");
27 var stepMessages = new List<string>();
28 stepMessages.Add("Steps contain errors:");
29 var arguments = new HashSet<string>() { "RandomSeed" };
30 var pipeline = GetGenerationPipeline();
31 var inputs = pipeline.GetGenerationInputs();
32 var steps = pipeline.GetGenerationSteps();
33
34 foreach (var input in inputs)
35 {
36 foreach (var name in input.OutputNames())
37 {
38 if (!arguments.Add(name))
39 inputMessages.Add($" * Duplicate input name: {name} <{input.GetType().Name}>.");
40 }
41 }
42
43 foreach (var step in steps)
44 {
45 foreach (var name in step.InputNames())
46 {
47 if (!arguments.Contains(name))
48 stepMessages.Add($" * Missing input name: {name} <{step.GetType().Name}>.");
49 }
50
51 foreach (var name in step.OutputNames())
52 {
53 arguments.Add(name);
54 }
55 }
56
57 if (inputMessages.Count > 1)
58 EditorGUILayout.HelpBox(string.Join('\n', inputMessages), MessageType.Error, true);
59 if (stepMessages.Count > 1)
60 EditorGUILayout.HelpBox(string.Join('\n', stepMessages), MessageType.Error, true);
61 }
