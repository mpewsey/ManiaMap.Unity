68 private static void DrawGenerationInputErrorBox(GenerationPipeline pipeline, HashSet<string> names)
69 {
70 var messages = new List<string>() { "Inputs contain errors:" };
71 var inputs = pipeline.FindInputComponents();
72
73 foreach (var input in inputs)
74 {
75 foreach (var name in input.OutputNames())
76 {
77 if (!names.Add(name))
78 messages.Add($" * Duplicate input name: {name} <{input.GetType().Name}>.");
79 }
80 }
81

83 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);

91 private static void DrawGenerationStepErrorBox(GenerationPipeline pipeline, HashSet<string> names)

96 foreach (var step in steps)

102 }
103
104 foreach (var name in step.OutputNames())

106 names.Add(name);
