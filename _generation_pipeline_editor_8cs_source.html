73 private void DrawGenerationInputErrorBox(HashSet<string> names)
74 {
75 var messages = new List<string>() { "Inputs contain errors:" };
76 var pipeline = GetGenerationPipeline();
77 var inputs = pipeline.GetGenerationInputs();
78
79 foreach (var input in inputs)
80 {
81 foreach (var name in input.OutputNames())
82 {
83 if (!names.Add(name))
84 messages.Add($" * Duplicate input name: {name} <{input.GetType().Name}>.");
85 }
86 }
87
88 if (messages.Count <= 1)
89 return;
90
91 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);
92 }

98 private void DrawGenerationStepErrorBox(HashSet<string> names)
99 {
100 var messages = new List<string>() { "Steps contain errors:" };
101 var pipeline = GetGenerationPipeline();
102 var steps = pipeline.GetGenerationSteps();
103
104 foreach (var step in steps)
105 {
106 foreach (var name in step.InputNames())
107 {
108 if (!names.Contains(name))
109 messages.Add($" * Missing input name: {name} <{step.GetType().Name}>.");
110 }
111
112 foreach (var name in step.OutputNames())
113 {
114 names.Add(name);
115 }
116 }
117
118 if (messages.Count <= 1)
119 return;
120
121 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);
