44 private void DrawGenerationInputErrorBox(HashSet<string> names)
45 {
46 var messages = new List<string>() { "Inputs contain errors:" };
47 var pipeline = GetGenerationPipeline();
48 var inputs = pipeline.GetGenerationInputs();
49
50 foreach (var input in inputs)

56 }

64 private void DrawGenerationStepErrorBox(HashSet<string> names)
65 {
66 var messages = new List<string>() { "Steps contain errors:" };
67 var pipeline = GetGenerationPipeline();
68 var steps = pipeline.GetGenerationSteps();
69
70 foreach (var step in steps)
71 {
72 foreach (var name in step.InputNames())
73 {
74 if (!names.Contains(name))
75 messages.Add($" * Missing input name: {name} <{step.GetType().Name}>.");
76 }
77
78 foreach (var name in step.OutputNames())
79 {
80 names.Add(name);
81 }
82 }
83
84 if (messages.Count > 1)
85 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);
86 }
