54 public async Task<PipelineResults> RunAttemptsAsync(int seed, int attempts = 10, int timeout = 5000, Dictionary<string, object> manualInputs = null, System.Action<string> logger = null)
55 {
56 manualInputs ??= new Dictionary<string, object>();
57
58 for (int i = 0; i < attempts; i++)
59 {
60 logger?.Invoke($"[Generation Pipeline] Beginning attempt {i + 1} / {attempts}...");
61 var inputs = new Dictionary<string, object>(manualInputs);
62 inputs.Add("RandomSeed", new RandomSeed(seed + i * 1447));
63 var token = new CancellationTokenSource(timeout).Token;
64 var results = await RunAsync(inputs, logger, token);
65
66 if (results.Success)
67 {
68 logger?.Invoke("[Generation Pipeline] Attempt successful.");
69 return results;
70 }
71 }
72
73 logger?.Invoke("[Generation Pipeline] Generation failed for all attempts.");
74 return new PipelineResults(manualInputs);
