260 public bool Paginate(Vector2 spacing)
261 {
262 var result = false;
263 var changed = true;
264
265 for (int k = 0; changed && k < 1000; k++)
266 {
267 changed = false;
268
269 for (int i = 0; i < Nodes.Count; i++)
270 {
271 for (int j = i + 1; j < Nodes.Count; j++)
272 {
273 var iNode = Nodes[i];
274 var jNode = Nodes[j];
275 var delta = jNode.Position - iNode.Position;
276
277 if (Mathf.Abs(delta.x) > spacing.x || Mathf.Abs(delta.y) > spacing.y)
278 continue;
279
280 var sign = new Vector2(Mathf.Sign(delta.x), Mathf.Sign(delta.y));
281 delta = spacing * sign - delta;
282
283 if (Mathf.Abs(delta.x) * spacing.y > Mathf.Abs(delta.y) * spacing.x)
284 delta.x = 0;
285 else
286 delta.y = 0;
287
288 if ((k & 1) == 0)
289 jNode.Position += delta;
290 else
291 iNode.Position -= delta;
292
293 changed = true;
294 }
295 }
296
297 result |= changed;
298 }
299
300 return result;
301 }
