107 private static List<RoomTemplate> SampleVariations()
108 {
109 var result = new List<RoomTemplate>();
110 var seed = new RandomSeed(12345);
111 int templateId = 1;
112 int collectableId = 1;
113
114 foreach (var template in SampleTemplates())
115 {
116 foreach (var variation in template.UniqueVariations())
117 {
118 var copy = new RoomTemplate(templateId++, variation.Name, variation.Cells);
119
120 foreach (var cell in copy.Cells.Array)
121 {
122 if (seed.NextDouble() <= 0.5)
123 cell?.AddCollectableSpot(collectableId++, "Default");
124 }
125
126 result.Add(copy);
127 }
128 }
129
130 return result;
131 }
