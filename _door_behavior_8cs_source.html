153 private DoorDirection FindClosestDirection()
154 {
155 var room = Room();
156 var delta = transform.position - Cell.transform.position;
157
158 System.Span<DoorDirection> directions = stackalloc DoorDirection[]
159 {
160 DoorDirection.North,
161 DoorDirection.South,
162 DoorDirection.East,
163 DoorDirection.West,
164 DoorDirection.Top,
165 DoorDirection.Bottom,
166 };
167
168 var distances = new float[]
169 {
170 Vector3.Dot(delta, room.Swizzle(Vector2.up)), // North
171 Vector3.Dot(delta, room.Swizzle(Vector2.down)), // South
172 Vector3.Dot(delta, room.Swizzle(Vector2.right)), // East
173 Vector3.Dot(delta, room.Swizzle(Vector2.left)), // West
174 Vector3.Dot(delta, room.Swizzle(Vector3.forward)), // Top
175 Vector3.Dot(delta, room.Swizzle(Vector3.back)), // Bottom
176 };
177
178 var max = Mathf.Max(distances);
179
180 for (int i = 0; i < distances.Length; i++)
181 {
182 if (distances[i] == max)
183 return directions[i];
184 }
185
186 return directions[0];
187 }
188 }
