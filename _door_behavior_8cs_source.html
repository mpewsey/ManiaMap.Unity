244 Vector3.Dot(delta, room.Swizzle(Vector2.down)), // South
245 Vector3.Dot(delta, room.Swizzle(Vector2.right)), // East
246 Vector3.Dot(delta, room.Swizzle(Vector2.left)), // West
247 Vector3.Dot(delta, room.Swizzle(Vector3.forward)), // Top
248 Vector3.Dot(delta, room.Swizzle(Vector3.back)), // Bottom
249 };
