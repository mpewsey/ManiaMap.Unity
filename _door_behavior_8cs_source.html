146 private DoorDirection FindClosestDirection()
147 {
148 var room = Room();
149 var delta = transform.position - Cell.transform.position;
150
151 System.Span<DoorDirection> directions = stackalloc DoorDirection[]
152 {
153 DoorDirection.North,
154 DoorDirection.South,
155 DoorDirection.East,
156 DoorDirection.West,
157 DoorDirection.Top,
158 DoorDirection.Bottom,
159 };
160
161 var distances = new float[]
162 {
163 Vector3.Dot(delta, room.Swizzle(Vector2.up)), // North
164 Vector3.Dot(delta, room.Swizzle(Vector2.down)), // South
165 Vector3.Dot(delta, room.Swizzle(Vector2.right)), // East
166 Vector3.Dot(delta, room.Swizzle(Vector2.left)), // West
167 Vector3.Dot(delta, room.Swizzle(Vector3.forward)), // Top
168 Vector3.Dot(delta, room.Swizzle(Vector3.back)), // Bottom
169 };
170
171 var max = Mathf.Max(distances);
172
173 for (int i = 0; i < distances.Length; i++)
174 {
175 if (distances[i] == max)
176 return directions[i];
177 }
178
179 return directions[0];
180 }
