239 Vector3.Dot(delta, room.Swizzle(Vector2.right)), // East
240 Vector3.Dot(delta, room.Swizzle(Vector2.left)), // West
241 Vector3.Dot(delta, room.Swizzle(Vector3.forward)), // Top
242 Vector3.Dot(delta, room.Swizzle(Vector3.back)), // Bottom
243 };
