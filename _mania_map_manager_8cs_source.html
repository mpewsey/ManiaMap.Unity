152 public void Initialize(Layout layout, LayoutState layoutState, ManiaMapSettings settings = null)

251 private void CreateRoomHandles(Uid id, RoomAddressableDatabase database)
252 {
253 foreach (var roomId in GetRoomCluster(id))
254 {
255 var room = Layout.Rooms[roomId];
256 var templateId = room.Template.Id;
257
258 if (!RoomHandles.TryGetValue(templateId, out var handle))
259 {
260 var prefab = database.GetPrefab(templateId);
261 handle = new RoomHandle(prefab.LoadAssetAsync());
262 RoomHandles.Add(templateId, handle);
263 }
264
265 handle.StaleCount = 0;
266 }
267 }
268
272 private void IncrementRoomHandles()
273 {
274 foreach (var handle in RoomHandles.Values)
275 {
276 handle.StaleCount++;
277 }
278 }
279
283 private void ReleaseStaleRoomHandles()
284 {
285 var maxStaleCount = Settings.MaxStaleCount;
286
287 foreach (var id in RoomHandles.Keys.ToList())
288 {
289 var handle = RoomHandles[id];
290
291 if (handle.StaleCount > maxStaleCount)
292 {
293 Addressables.Release(handle.Handle);
294 RoomHandles.Remove(id);
295 }
296 }
