62 public static void DrawNodeTemplateGroupErrorBox(LayoutGraphObject graph)
63 {
64 var messages = new List<string>() { "Nodes are missing template groups:" };
65
66 foreach (var node in graph.GetNodes().OrderBy(x => x.Id))
67 {
68 if (node.TemplateGroup == null)
69 messages.Add($" * {node.Id} : {node.Name}");
70 }
71
72 if (messages.Count <= 1)
73 return;
74
75 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);
76 }

82 public static void DrawEdgeTemplateGroupErrorBox(LayoutGraphObject graph)
83 {
84 var messages = new List<string>() { "Edges with non-zero room chances are missing template groups:" };
85
86 foreach (var edge in graph.GetEdges().OrderBy(x => new EdgeIndexes(x.FromNode, x.ToNode)))
87 {
88 if (edge.RoomChance > 0 && edge.TemplateGroup == null)
89 messages.Add($" * ({edge.FromNode}, {edge.ToNode}) : {edge.Name}");
90 }
91
92 if (messages.Count <= 1)
93 return;
94
95 EditorGUILayout.HelpBox(string.Join('\n', messages), MessageType.Error, true);
