281 public bool Paginate(Vector2 spacing)
282 {
283 var result = false;
284 var changed = true;
285
286 for (int k = 0; changed && k < 1000; k++)
287 {
288 changed = false;
289
290 for (int i = 0; i < Nodes.Count; i++)
291 {
292 for (int j = i + 1; j < Nodes.Count; j++)
293 {
294 var iNode = Nodes[i];
295 var jNode = Nodes[j];
296 var delta = jNode.Position - iNode.Position;
297
298 if (Mathf.Abs(delta.x) > spacing.x || Mathf.Abs(delta.y) > spacing.y)
299 continue;
300
301 var sign = new Vector2(Mathf.Sign(delta.x), Mathf.Sign(delta.y));
302 delta = spacing * sign - delta;
303
304 if (Mathf.Abs(delta.x) * spacing.y > Mathf.Abs(delta.y) * spacing.x)
305 delta.x = 0;
306 else
307 delta.y = 0;
308
309 if ((k & 1) == 0)
310 jNode.Position += delta;
311 else
312 iNode.Position -= delta;
313
314 changed = true;
315 }
316 }
317
318 result |= changed;
319 }
320
321 return result;
322 }
