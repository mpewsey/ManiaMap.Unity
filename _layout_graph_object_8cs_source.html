282 public bool Paginate(Vector2 spacing)
283 {
284 var result = false;
285 var changed = true;
286
287 for (int k = 0; changed && k < 1000; k++)
288 {
289 changed = false;
290
291 for (int i = 0; i < Nodes.Count; i++)
292 {
293 for (int j = i + 1; j < Nodes.Count; j++)
294 {
295 var iNode = Nodes[i];
296 var jNode = Nodes[j];
297 var delta = jNode.Position - iNode.Position;
298
299 if (Mathf.Abs(delta.x) > spacing.x || Mathf.Abs(delta.y) > spacing.y)
300 continue;
301
302 var sign = new Vector2(Mathf.Sign(delta.x), Mathf.Sign(delta.y));
303 delta = spacing * sign - delta;
304
305 if (Mathf.Abs(delta.x) * spacing.y > Mathf.Abs(delta.y) * spacing.x)
306 delta.x = 0;
307 else
308 delta.y = 0;
309
310 if ((k & 1) == 0)
311 jNode.Position += delta;
312 else
313 iNode.Position -= delta;
314
315 changed = true;
316 }
317 }
318
319 result |= changed;
320 }
321
322 return result;
323 }
