115 public static AsyncOperationHandle<GameObject> InstantiateRoomAsync(Uid id,
116 AssetReferenceGameObject prefab, Transform parent = null,
117 RoomPositionOption position = RoomPositionOption.UseManagerSettings)
118 {
119 var manager = ManiaMapManager.CurrentValidated;
120 position = manager.Settings.GetRoomPositionOption(position);
121 var layout = manager.Layout;
122 var layoutState = manager.LayoutState;
123 var doorConnections = manager.GetDoorConnections(id);
124 var handle = prefab.InstantiateAsync(parent);
125
126 handle.Completed += x =>
127 {
128 if (!x.IsValid() || !x.IsDone || x.Result == null)
129 throw new InstantiationFailedException("Failed to instantiate room.");
130
131 if (!x.Result.TryGetComponent(out Room room))
132 throw new MissingRoomComponentException($"Prefab does not have room component: {x.Result}.");
133
134 room.Initialize(id, layout, layoutState, doorConnections, position);
135 };
136
137 return handle;
138 }

244 public Vector3 Swizzle(Vector3 vector)
245 {
246 switch (CellPlane)
247 {
248 case Plane.XY:
249 return new Vector3(vector.x, vector.y, -vector.z);
250 case Plane.XZ:
251 return new Vector3(vector.x, vector.z, vector.y);
252 default:
253 throw new System.ArgumentException($"Unhandled cell plane: {CellPlane}.");
254 }

327 public void CreateCells()
328 {
329 CreateCellContainer();
330
331 // Destroy extra rows.
332 while (CellContainer.childCount > Size.x)
333 {
334 var row = CellContainer.GetChild(CellContainer.childCount - 1);
335 DestroyImmediate(row.gameObject);
336 }
337
338 // Create new rows.
339 while (CellContainer.childCount < Size.x)
340 {
341 var obj = new GameObject("<New Row>");
342 obj.transform.SetParent(CellContainer);
343 }
344
345 for (int i = 0; i < CellContainer.childCount; i++)
346 {
347 CreateRow(i);
348 }
349 }

356 private void CreateRow(int row)
357 {
358 var container = CellContainer.GetChild(row);
359 container.name = $"<Row {row}>";
360
361 // Destroy extra cells.
362 while (container.childCount > Size.y)
363 {
364 var cell = container.GetChild(container.childCount - 1);
365 DestroyImmediate(cell.gameObject);
366 }
367
368 // Create new cells.
369 while (container.childCount < Size.y)
370 {
371 var obj = new GameObject("<New Cell>");
372 obj.transform.SetParent(container);
373 obj.AddComponent<Cell>();
374 }
375
376 for (int j = 0; j < container.childCount; j++)
377 {
378 var index = new Vector2Int(row, j);
379 var cell = container.GetChild(j).GetComponent<Cell>();
380 cell.Initialize(this, index);
381 }

387 public ManiaMap.RoomTemplate GetTemplate()
