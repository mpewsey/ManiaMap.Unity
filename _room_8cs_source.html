250 public void CreateCells()
251 {
252 CreateCellContainer();
253
254 // Destroy extra rows.
255 while (CellContainer.childCount > Size.x)
256 {
257 var row = CellContainer.GetChild(CellContainer.childCount - 1);
258 DestroyImmediate(row.gameObject);
259 }
260
261 // Create new rows.
262 while (CellContainer.childCount < Size.x)
263 {
264 var obj = new GameObject("<New Row>");
265 obj.transform.SetParent(CellContainer);
266 }
267
268 for (int i = 0; i < CellContainer.childCount; i++)
269 {
270 CreateRow(i);
271 }
272 }

279 private void CreateRow(int row)
280 {
281 var container = CellContainer.GetChild(row);
282 container.name = $"<Row {row}>";
283
284 // Destroy extra cells.
285 while (container.childCount > Size.y)
286 {
287 var cell = container.GetChild(container.childCount - 1);
288 DestroyImmediate(cell.gameObject);
289 }
290
291 // Create new cells.
292 while (container.childCount < Size.y)
293 {
294 var obj = new GameObject("<New Cell>");
295 obj.transform.SetParent(container);
296 obj.AddComponent<Cell>();
297 }
298
299 for (int j = 0; j < container.childCount; j++)
300 {
301 var index = new Vector2Int(row, j);
302 var cell = container.GetChild(j).GetComponent<Cell>();
303 cell.Init(this, index);
304 }
305 }
