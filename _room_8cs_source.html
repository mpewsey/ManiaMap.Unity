177 public void CreateCells()
178 {
179 CreateCellContainer();
180
181 // Destroy extra rows.
182 while (CellContainer.childCount > Size.x)
183 {
184 var row = CellContainer.GetChild(CellContainer.childCount - 1);
185 DestroyImmediate(row.gameObject);
186 }
187
188 // Create new rows.
189 while (CellContainer.childCount < Size.x)
190 {
191 var obj = new GameObject("<New Row>");
192 obj.transform.SetParent(CellContainer);
193 }
194
195 for (int i = 0; i < CellContainer.childCount; i++)
196 {
197 CreateRow(i);
198 }
199 }

206 private void CreateRow(int row)
207 {
208 var container = CellContainer.GetChild(row);
209 container.name = $"<Row {row}>";
210
211 // Destroy extra cells.
212 while (container.childCount > Size.y)
213 {
214 var cell = container.GetChild(container.childCount - 1);
215 DestroyImmediate(cell.gameObject);
216 }
217
218 // Create new cells.
219 while (container.childCount < Size.y)
220 {
221 var obj = new GameObject("<New Cell>");
222 obj.transform.SetParent(container);
223 obj.AddComponent<Cell>();
224 }
225
226 for (int j = 0; j < container.childCount; j++)
227 {
228 var index = new Vector2Int(row, j);
229 var cell = container.GetChild(j).GetComponent<Cell>();
230 cell.Init(this, index);
231 }
