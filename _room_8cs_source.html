129 private static AsyncOperationHandle<GameObject> InstantiateRoomHandle(Uid id, AssetReferenceGameObject prefab, Transform parent, RoomPositionOption position = RoomPositionOption.Default)
130 {
131 var handle = Addressables.InstantiateAsync(prefab, parent);
132
133 handle.Completed += x =>
134 {
135 if (!x.IsValid() || !x.IsDone || x.Result == null)
136 throw new System.ArgumentException("Failed to instantiate room.");
137 if (!x.Result.TryGetComponent(out Room room))
138 throw new System.ArgumentException($"Prefab does not have room component: {x.Result}.");
139 room.Initialize(id, position);
140 };
141
142 return handle;
143 }

275 {

323 public void CreateCells()
324 {
325 CreateCellContainer();
326
327 // Destroy extra rows.
328 while (CellContainer.childCount > Size.x)
329 {
330 var row = CellContainer.GetChild(CellContainer.childCount - 1);
331 DestroyImmediate(row.gameObject);
332 }
333
334 // Create new rows.
335 while (CellContainer.childCount < Size.x)
336 {
337 var obj = new GameObject("<New Row>");
338 obj.transform.SetParent(CellContainer);
339 }
340
341 for (int i = 0; i < CellContainer.childCount; i++)
342 {
343 CreateRow(i);
344 }
345 }

352 private void CreateRow(int row)
353 {
354 var container = CellContainer.GetChild(row);
355 container.name = $"<Row {row}>";
356
357 // Destroy extra cells.
358 while (container.childCount > Size.y)
359 {
360 var cell = container.GetChild(container.childCount - 1);
361 DestroyImmediate(cell.gameObject);
362 }
363
364 // Create new cells.
365 while (container.childCount < Size.y)
366 {
367 var obj = new GameObject("<New Cell>");
368 obj.transform.SetParent(container);
369 obj.AddComponent<Cell>();
370 }
371
372 for (int j = 0; j < container.childCount; j++)
373 {
374 var index = new Vector2Int(row, j);
375 var cell = container.GetChild(j).GetComponent<Cell>();
376 cell.Initialize(this, index);
377 }
