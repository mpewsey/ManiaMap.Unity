248 public void CreateCells()
249 {
250 CreateCellContainer();
251
252 // Destroy extra rows.
253 while (CellContainer.childCount > Size.x)
254 {
255 var row = CellContainer.GetChild(CellContainer.childCount - 1);
256 DestroyImmediate(row.gameObject);
257 }
258
259 // Create new rows.
260 while (CellContainer.childCount < Size.x)
261 {
262 var obj = new GameObject("<New Row>");
263 obj.transform.SetParent(CellContainer);
264 }
265
266 for (int i = 0; i < CellContainer.childCount; i++)
267 {
268 CreateRow(i);
269 }
270 }
271
277 private void CreateRow(int row)
278 {
279 var container = CellContainer.GetChild(row);
280 container.name = $"<Row {row}>";
281
282 // Destroy extra cells.
283 while (container.childCount > Size.y)
284 {
285 var cell = container.GetChild(container.childCount - 1);
286 DestroyImmediate(cell.gameObject);
287 }
288
289 // Create new cells.
290 while (container.childCount < Size.y)
291 {
292 var obj = new GameObject("<New Cell>");
293 obj.transform.SetParent(container);
294 obj.AddComponent<Cell>();
295 }
296
297 for (int j = 0; j < container.childCount; j++)
298 {
299 var index = new Vector2Int(row, j);
300 var cell = container.GetChild(j).GetComponent<Cell>();
301 cell.Init(this, index);
302 }
303 }

308 public RoomTemplate GetTemplate()
309 {
310 AutoAssign();

312
