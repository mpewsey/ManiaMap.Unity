184 public void CreateCells()
185 {
186 CreateCellContainer();
187
188 // Destroy extra rows.
189 while (CellContainer.childCount > Size.x)
190 {
191 var row = CellContainer.GetChild(CellContainer.childCount - 1);
192 DestroyImmediate(row.gameObject);
193 }
194
195 // Create new rows.
196 while (CellContainer.childCount < Size.x)
197 {
198 var obj = new GameObject("<New Row>");
199 obj.transform.SetParent(CellContainer);
200 }
201
202 for (int i = 0; i < CellContainer.childCount; i++)

204 CreateRow(i);

213 private void CreateRow(int row)
214 {

218 // Destroy extra cells.
219 while (container.childCount > Size.y)

221 var cell = container.GetChild(container.childCount - 1);
222 DestroyImmediate(cell.gameObject);
223 }
224
225 // Create new cells.
226 while (container.childCount < Size.y)
227 {
228 var obj = new GameObject("<New Cell>");
229 obj.transform.SetParent(container);
230 obj.AddComponent<Cell>();
231 }
232
233 for (int j = 0; j < container.childCount; j++)
234 {
235 var index = new Vector2Int(row, j);
236 var cell = container.GetChild(j).GetComponent<Cell>();
237 cell.Init(this, index);
