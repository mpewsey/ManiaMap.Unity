108 public static AsyncOperationHandle<GameObject> InstantiateRoomAsync(Uid id, AssetReferenceGameObject prefab, Transform parent = null, RoomPositionOption position = RoomPositionOption.Default)
109 {
110 var layout = ManiaMapManager.Current.Layout;
111 var layoutState = ManiaMapManager.Current.LayoutState;
112 var handle = prefab.InstantiateAsync(parent);
113
114 handle.Completed += x =>
115 {
116 if (!x.IsValid() || !x.IsDone || x.Result == null)
117 throw new InstantiationFailedException("Failed to instantiate room.");
118
119 if (!x.Result.TryGetComponent(out Room room))
120 throw new MissingRoomComponentException($"Prefab does not have room component: {x.Result}.");
121
122 room.Initialize(id, layout, layoutState, position);
123 };
124
125 return handle;
126 }

314 public void CreateCells()
315 {
316 CreateCellContainer();
317
318 // Destroy extra rows.
319 while (CellContainer.childCount > Size.x)
320 {
321 var row = CellContainer.GetChild(CellContainer.childCount - 1);
322 DestroyImmediate(row.gameObject);
323 }
324
325 // Create new rows.
326 while (CellContainer.childCount < Size.x)
327 {
328 var obj = new GameObject("<New Row>");
329 obj.transform.SetParent(CellContainer);
330 }
331
332 for (int i = 0; i < CellContainer.childCount; i++)
333 {
334 CreateRow(i);
335 }
336 }

343 private void CreateRow(int row)
344 {
345 var container = CellContainer.GetChild(row);
346 container.name = $"<Row {row}>";
347
348 // Destroy extra cells.
349 while (container.childCount > Size.y)
350 {
351 var cell = container.GetChild(container.childCount - 1);
352 DestroyImmediate(cell.gameObject);
353 }
354
355 // Create new cells.
356 while (container.childCount < Size.y)
357 {
358 var obj = new GameObject("<New Cell>");
359 obj.transform.SetParent(container);
360 obj.AddComponent<Cell>();
361 }
362
363 for (int j = 0; j < container.childCount; j++)
364 {
365 var index = new Vector2Int(row, j);
366 var cell = container.GetChild(j).GetComponent<Cell>();
367 cell.Initialize(this, index);
368 }
