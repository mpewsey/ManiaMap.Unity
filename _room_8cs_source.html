260 public void CreateCells()
261 {
262 CreateCellContainer();
263
264 // Destroy extra rows.
265 while (CellContainer.childCount > Size.x)
266 {
267 var row = CellContainer.GetChild(CellContainer.childCount - 1);
268 DestroyImmediate(row.gameObject);
269 }
270
271 // Create new rows.
272 while (CellContainer.childCount < Size.x)
273 {
274 var obj = new GameObject("<New Row>");
275 obj.transform.SetParent(CellContainer);
276 }
277
278 for (int i = 0; i < CellContainer.childCount; i++)
279 {
280 CreateRow(i);
281 }
282 }

289 private void CreateRow(int row)
290 {
291 var container = CellContainer.GetChild(row);
292 container.name = $"<Row {row}>";
293
294 // Destroy extra cells.
295 while (container.childCount > Size.y)
296 {
297 var cell = container.GetChild(container.childCount - 1);
298 DestroyImmediate(cell.gameObject);
299 }
300
301 // Create new cells.
302 while (container.childCount < Size.y)
303 {
304 var obj = new GameObject("<New Cell>");
305 obj.transform.SetParent(container);
306 obj.AddComponent<Cell>();
307 }
308
309 for (int j = 0; j < container.childCount; j++)
310 {
311 var index = new Vector2Int(row, j);
312 var cell = container.GetChild(j).GetComponent<Cell>();
313 cell.Init(this, index);
314 }

320 public ManiaMap.RoomTemplate GetTemplate()
