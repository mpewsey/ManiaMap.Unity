153 private void EnsureCapacity(int capacity)
154 {
155 CreateGrid();
156
157 // Destroy extra layers.
158 while (Layers.Count > capacity)
159 {
160 var index = Layers.Count - 1;
161 Destroy(Layers[index].gameObject);
162 Layers.RemoveAt(index);
163 }
164
165 // Create missing layers.
166 while (Layers.Count < capacity)
167 {
168 Layers.Add(LayoutTilemapLayer.Create(this));
169 }
170 }

177 private void DrawMap(Tilemap tilemap, int z)
178 {
179 tilemap.ClearAllTiles();
180
181 foreach (var room in Layout.Rooms.Values)
182 {
183 // If room Z (layer) value is not equal, go to next room.
184 if (room.Position.Z != z)
185 continue;
186
187 var roomState = LayoutState?.RoomStates[room.Id];
188 var cells = room.Template.Cells;
189
190 for (int i = 0; i < cells.Rows; i++)
191 {
192 for (int j = 0; j < cells.Columns; j++)
193 {
194 var cell = cells[i, j];
195 var position = new Vector2DInt(i, j);
196
197 // If cell it empty, go to next cell.
198 if (cell == null)
199 continue;
200
201 // If room state is defined and is not visible, go to next cell.
202 if (roomState != null && !roomState.IsVisible && !roomState.CellIsVisible(position))
203 continue;
204
205 // Calculate draw position
206 var x = room.Position.Y + j;
207 var y = -room.Position.X - i;
208 var point = new Vector3Int(x, y, 0);
209
210 // Get adjacent cells
211 var north = cells.GetOrDefault(i - 1, j);
212 var south = cells.GetOrDefault(i + 1, j);
213 var west = cells.GetOrDefault(i, j - 1);
214 var east = cells.GetOrDefault(i, j + 1);
215
216 // Accumulate map tile types
217 var flags = GetFeatureFlags(cell);
218 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Top);
219 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
220 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
221 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
222 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
223 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
224
225 // Set the map tile.
226 var color = roomState == null || roomState.CellIsVisible(position) ? ColorUtility.ConvertColor(room.Color) : RoomColor;
227 var tile = MapTilePool.GetTile(flags, color);
228 tilemap.SetTile(point, tile);
229 }
230 }
231 }
232 }

258 private long GetTileFlag(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
