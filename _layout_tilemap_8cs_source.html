188 private List<LayoutTilemapLayer> CreateLayerComponents()
189 {
190 CreateGrid();
191 var layers = Grid.GetComponentsInChildren<LayoutTilemapLayer>().ToList();
192 var zs = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
193
194 // Destroy extra layers.
195 for (int i = layers.Count - 1; i >= 0; i--)
196 {
197 var layer = layers[i];
198
199 if (!zs.Contains(layer.Z))
200 {
201 Destroy(layer.gameObject);
202 layers.RemoveAt(i);
203 }
204 }
205
206 // Create missing layers.
207 foreach (var z in zs)
208 {
209 if (!layers.Any(x => x.Z == z))
210 {
211 layers.Add(LayoutTilemapLayer.Create(this, z));
212 }
213 }
214
215 return layers;
216 }

223 private void DrawMap(Tilemap tilemap, int z)
224 {
225 tilemap.ClearAllTiles();
226
227 foreach (var room in Layout.Rooms.Values)
228 {
229 // If room Z (layer) value is not equal, go to next room.
230 if (room.Position.Z != z)
231 continue;
232
233 var roomState = LayoutState?.RoomStates[room.Id];
234 var cells = room.Template.Cells;
235
236 for (int i = 0; i < cells.Rows; i++)
237 {
238 for (int j = 0; j < cells.Columns; j++)
239 {
240 var cell = cells[i, j];
241 var position = new Vector2DInt(i, j);
242
243 // If cell it empty, go to next cell.
244 if (cell == null)
245 continue;
246
247 // If room state is defined and is not visible, go to next cell.
248 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
249 continue;
250
251 // Calculate draw position
252 var x = room.Position.Y + j;
253 var y = -room.Position.X - i;
254 var point = new Vector3Int(x, y, 0);
255
256 // Get adjacent cells
257 var north = cells.GetOrDefault(i - 1, j);
258 var south = cells.GetOrDefault(i + 1, j);
259 var west = cells.GetOrDefault(i, j - 1);
260 var east = cells.GetOrDefault(i, j + 1);
261
262 // Accumulate map tile types
263 var tileTypes = MapTileTypes.None;
264 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Top);
265 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Bottom);
266 tileTypes |= GetTileType(room, cell, north, position, DoorDirection.North);
267 tileTypes |= GetTileType(room, cell, south, position, DoorDirection.South);
268 tileTypes |= GetTileType(room, cell, west, position, DoorDirection.West);
269 tileTypes |= GetTileType(room, cell, east, position, DoorDirection.East);
270
271 // Set the map tile.
272 var tile = GetTile(tileTypes, ColorUtility.ConvertColor(room.Color));
273 tilemap.SetTile(point, tile);
274 }
275 }
276 }
277 }

287 private MapTileTypes GetTileType(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
288 {
