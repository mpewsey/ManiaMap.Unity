196 private List<LayoutTilemapLayer> CreateLayerComponents()
197 {
198 CreateGrid();
199 var layers = Grid.GetComponentsInChildren<LayoutTilemapLayer>().ToList();
200 var zs = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
201
202 // Destroy extra layers.
203 for (int i = layers.Count - 1; i >= 0; i--)
204 {
205 var layer = layers[i];
206
207 if (!zs.Contains(layer.Z))
208 {
209 Destroy(layer.gameObject);
210 layers.RemoveAt(i);
211 }
212 }
213
214 // Create missing layers.
215 foreach (var z in zs)
216 {
217 if (!layers.Any(x => x.Z == z))
218 {
219 layers.Add(LayoutTilemapLayer.Create(this, z));
220 }
221 }
222
223 return layers;
224 }

231 private void DrawMap(Tilemap tilemap, int z)
232 {
233 tilemap.ClearAllTiles();
234
235 foreach (var room in Layout.Rooms.Values)
236 {
237 // If room Z (layer) value is not equal, go to next room.
238 if (room.Position.Z != z)
239 continue;
240
241 var roomState = LayoutState?.RoomStates[room.Id];
242 var cells = room.Template.Cells;
243
244 for (int i = 0; i < cells.Rows; i++)
245 {
246 for (int j = 0; j < cells.Columns; j++)
247 {
248 var cell = cells[i, j];
249 var position = new Vector2DInt(i, j);
250
251 // If cell it empty, go to next cell.
252 if (cell == null)
253 continue;
254
255 // If room state is defined and is not visible, go to next cell.
256 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
257 continue;
258
259 // Calculate draw position
260 var x = room.Position.Y + j;
261 var y = -room.Position.X - i;
262 var point = new Vector3Int(x, y, 0);
263
264 // Get adjacent cells
265 var north = cells.GetOrDefault(i - 1, j);
266 var south = cells.GetOrDefault(i + 1, j);
267 var west = cells.GetOrDefault(i, j - 1);
268 var east = cells.GetOrDefault(i, j + 1);
269
270 // Accumulate map tile types
271 var tileTypes = MapTileTypes.None;
272 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Top);
273 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Bottom);
274 tileTypes |= GetTileType(room, cell, north, position, DoorDirection.North);
275 tileTypes |= GetTileType(room, cell, south, position, DoorDirection.South);
276 tileTypes |= GetTileType(room, cell, west, position, DoorDirection.West);
277 tileTypes |= GetTileType(room, cell, east, position, DoorDirection.East);
278
279 // Set the map tile.
280 var tile = GetTile(tileTypes, ColorUtility.ConvertColor(room.Color));
281 tilemap.SetTile(point, tile);
282 }
283 }
284 }
285 }
