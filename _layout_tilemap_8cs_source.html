130 private void DrawMapTiles(Tilemap tilemap, int z)
131 {
132 tilemap.ClearAllTiles();
133
134 foreach (var room in Layout.Rooms.Values)
135 {
136 // If room Z (layer) value is not equal, go to next room.
137 if (room.Position.Z != z)
138 continue;
139
140 var roomState = LayoutState?.RoomStates[room.Id];
141 var cells = room.Template.Cells;
142
143 for (int i = 0; i < cells.Rows; i++)
144 {
145 for (int j = 0; j < cells.Columns; j++)
146 {
147 var cell = cells[i, j];
148 var position = new Vector2DInt(i, j);
149
150 // If cell it empty, go to next cell.
151 if (cell == null)
152 continue;
153
154 // If room state is defined and is not visible, go to next cell.
155 if (roomState != null && !roomState.VisibleIndexes.Contains(position))
156 continue;
157
158 // Calculate draw position
159 var x = room.Position.Y + j;
160 var y = -room.Position.X - i;
161 var point = new Vector3Int(x, y, 0);
162
163 // Get adjacent cells
164 var north = cells.GetOrDefault(i - 1, j);
165 var south = cells.GetOrDefault(i + 1, j);
166 var west = cells.GetOrDefault(i, j - 1);
167 var east = cells.GetOrDefault(i, j + 1);
168
169 var tileTypes = MapTileTypes.None;
170 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Top);
171 tileTypes |= GetTileType(room, cell, null, position, DoorDirection.Bottom);
172 tileTypes |= GetTileType(room, cell, north, position, DoorDirection.North);
173 tileTypes |= GetTileType(room, cell, south, position, DoorDirection.South);
174 tileTypes |= GetTileType(room, cell, west, position, DoorDirection.West);
175 tileTypes |= GetTileType(room, cell, east, position, DoorDirection.East);
176
177 var tile = GetTile(tileTypes, ColorUtility.ConvertColor(room.Color));
178 tilemap.SetTile(point, tile);
179 }
180 }
181 }
182 }

184 private MapTileTypes GetTileType(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)
185 {
186 if (cell.GetDoor(direction) != null && DoorExists(room, position, direction))
