105 public void CreateLayers(Layout layout, LayoutState state)
106 {
107 Initialize(layout, state);
108 var zs = new HashSet<int>(Layout.Rooms.Values.Select(x => x.Position.Z));
109 EnsureCapacity(zs.Count);
110
111 // Draw the layers.
112 int i = 0;
113 var layers = GetComponentsInChildren<LayoutTilemapLayer>();
114
115 foreach (var z in zs.OrderBy(x => x))
116 {
117 var layer = layers[i++];
118 layer.Initialize(z);
119 DrawMap(layer.Tilemap, layer.Z);
120 }
121 }

127 private void EnsureCapacity(int capacity)
128 {
129 CreateGrid();

132 // Destroy extra layers.
133 while (container.childCount > capacity)

150 private void DrawMap(Tilemap tilemap, int z)
151 {
152 tilemap.ClearAllTiles();
153
154 foreach (var room in Layout.Rooms.Values)
155 {
156 // If room Z (layer) value is not equal, go to next room.
157 if (room.Position.Z != z)
158 continue;
159
160 var roomState = LayoutState?.RoomStates[room.Id];
161 var cells = room.Template.Cells;
162
163 for (int i = 0; i < cells.Rows; i++)
164 {
165 for (int j = 0; j < cells.Columns; j++)
166 {
167 var cell = cells[i, j];
168 var position = new Vector2DInt(i, j);
169
170 // If cell it empty, go to next cell.
171 if (cell == null)
172 continue;
173
174 // If room state is defined and is not visible, go to next cell.
175 if (roomState != null && !roomState.CellIsVisible(position))
176 continue;
177
178 // Calculate draw position
179 var x = room.Position.Y + j;
180 var y = -room.Position.X - i;
181 var point = new Vector3Int(x, y, 0);
182
183 // Get adjacent cells
184 var north = cells.GetOrDefault(i - 1, j);
185 var south = cells.GetOrDefault(i + 1, j);
186 var west = cells.GetOrDefault(i, j - 1);
187 var east = cells.GetOrDefault(i, j + 1);
188
189 // Accumulate map tile types
190 var flags = GetFeatureFlags(cell);
191 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Top);
192 flags |= GetTileFlag(room, cell, null, position, DoorDirection.Bottom);
193 flags |= GetTileFlag(room, cell, north, position, DoorDirection.North);
194 flags |= GetTileFlag(room, cell, south, position, DoorDirection.South);
195 flags |= GetTileFlag(room, cell, west, position, DoorDirection.West);
196 flags |= GetTileFlag(room, cell, east, position, DoorDirection.East);
197
198 // Set the map tile.
199 var tile = MapTilePool.GetTile(flags, ColorUtility.ConvertColor(room.Color));
200 tilemap.SetTile(point, tile);
201 }
202 }
203 }

230 private long GetTileFlag(ManiaMap.Room room, ManiaMap.Cell cell, ManiaMap.Cell neighbor, Vector2DInt position, DoorDirection direction)

232 if (cell.GetDoor(direction) != null && DoorExists(room, position, direction))
